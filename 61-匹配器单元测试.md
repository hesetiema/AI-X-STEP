import {
  collectNetworkEvidence,
  getNetworkEvidence,
  isGatewayLikeEvidence,
  isHttp5xxEvidence,
  isTimeoutEvidence,
} from '../../../../src/diagnosis/domain/rule-engine/matchers/network.matcher';
import { buildDiagnosisContext } from '../../../helpers/diagnosis-context.fixture';

describe('network.matcher', () => {
  describe('getNetworkEvidence', () => {
    it('should return only network_error evidence', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-net-1',
            type: 'network_error',
            label: 'POST /orders 500',
            value: { status: 500 },
            source: 'network-observer',
          },
          {
            id: 'ev-ui-1',
            type: 'ui_state',
            label: 'loading > 10s',
            value: { loadingMs: 10000 },
            source: 'frontend-sdk',
          },
          {
            id: 'ev-net-2',
            type: 'network_error',
            label: 'gateway timeout',
            value: { status: 504 },
            source: 'network-observer',
          },
        ],
      });

      const result = getNetworkEvidence(context);

      expect(result.map((x) => x.id)).toEqual(['ev-net-1', 'ev-net-2']);
    });
  });

  describe('isHttp5xxEvidence', () => {
    it('should return true for 5xx status', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-1',
            type: 'network_error',
            label: 'POST /orders 500',
            value: { status: 500 },
            source: 'network-observer',
          },
        ],
      });

      expect(isHttp5xxEvidence(context.evidence[0])).toBe(true);
    });

    it('should return false for non-5xx status', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-2',
            type: 'network_error',
            label: 'POST /orders 400',
            value: { status: 400 },
            source: 'network-observer',
          },
        ],
      });

      expect(isHttp5xxEvidence(context.evidence[0])).toBe(false);
    });

    it('should return false when status is missing', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-3',
            type: 'network_error',
            label: 'unknown error',
            value: {},
            source: 'network-observer',
          },
        ],
      });

      expect(isHttp5xxEvidence(context.evidence[0])).toBe(false);
    });
  });

  describe('isTimeoutEvidence', () => {
    it('should return true when label contains timeout', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-timeout-1',
            type: 'network_error',
            label: 'POST /orders timeout',
            value: { durationMs: 1000 },
            source: 'network-observer',
          },
        ],
      });

      expect(isTimeoutEvidence(context.evidence[0])).toBe(true);
    });

    it('should return true when code is ETIMEDOUT', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-timeout-2',
            type: 'network_error',
            label: 'request failed',
            value: { code: 'ETIMEDOUT' },
            source: 'network-observer',
          },
        ],
      });

      expect(isTimeoutEvidence(context.evidence[0])).toBe(true);
    });

    it('should return true when duration exceeds threshold', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-timeout-3',
            type: 'network_error',
            label: 'slow request',
            value: { durationMs: 5000 },
            source: 'network-observer',
          },
        ],
      });

      expect(
        isTimeoutEvidence(context.evidence[0], { minDurationMs: 3000 }),
      ).toBe(true);
    });

    it('should return false when duration is below threshold and no timeout signal exists', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-timeout-4',
            type: 'network_error',
            label: 'slow request',
            value: { durationMs: 1200 },
            source: 'network-observer',
          },
        ],
      });

      expect(
        isTimeoutEvidence(context.evidence[0], { minDurationMs: 3000 }),
      ).toBe(false);
    });
  });

  describe('isGatewayLikeEvidence', () => {
    it('should return true when label contains gateway', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-gw-1',
            type: 'network_error',
            label: 'gateway bad gateway',
            value: {},
            source: 'network-observer',
          },
        ],
      });

      expect(isGatewayLikeEvidence(context.evidence[0])).toBe(true);
    });

    it('should return true when gateway flag is true', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-gw-2',
            type: 'network_error',
            label: 'request failed',
            value: { gateway: true },
            source: 'network-observer',
          },
        ],
      });

      expect(isGatewayLikeEvidence(context.evidence[0])).toBe(true);
    });

    it('should return true for 502 or 504 status', () => {
      const context502 = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-gw-3',
            type: 'network_error',
            label: 'bad gateway',
            value: { status: 502 },
            source: 'network-observer',
          },
        ],
      });

      const context504 = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-gw-4',
            type: 'network_error',
            label: 'gateway timeout',
            value: { status: 504 },
            source: 'network-observer',
          },
        ],
      });

      expect(isGatewayLikeEvidence(context502.evidence[0])).toBe(true);
      expect(isGatewayLikeEvidence(context504.evidence[0])).toBe(true);
    });

    it('should return false for normal non-gateway request', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-gw-5',
            type: 'network_error',
            label: 'POST /orders 400',
            value: { status: 400 },
            source: 'network-observer',
          },
        ],
      });

      expect(isGatewayLikeEvidence(context.evidence[0])).toBe(false);
    });
  });

  describe('collectNetworkEvidence', () => {
    it('should collect only matched network evidence', () => {
      const context = buildDiagnosisContext({
        evidence: [
          {
            id: 'ev-1',
            type: 'network_error',
            label: 'POST /orders 500',
            value: { status: 500 },
            source: 'network-observer',
          },
          {
            id: 'ev-2',
            type: 'network_error',
            label: 'POST /orders 400',
            value: { status: 400 },
            source: 'network-observer',
          },
          {
            id: 'ev-3',
            type: 'ui_state',
            label: 'loading > 10s',
            value: { loadingMs: 10000 },
            source: 'frontend-sdk',
          },
        ],
      });

      const result = collectNetworkEvidence(context, isHttp5xxEvidence);

      expect(result.map((x) => x.id)).toEqual(['ev-1']);
    });
  });
});
