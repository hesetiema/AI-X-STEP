## 3.1 infrastructure/diagnosis.repository.ts

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisTaskEntity } from '../models/diagnosis-task.entity';
import { RawEvidenceBundle } from '../models/raw-evidence.model';

@Injectable()
export class DiagnosisRepository {
  private readonly taskStore = new Map<string, DiagnosisTaskEntity>();
  private readonly evidenceStore = new Map<string, RawEvidenceBundle>();

  async create(task: DiagnosisTaskEntity): Promise<void> {
    this.taskStore.set(task.diagnosisId, task);
  }

  async findById(diagnosisId: string): Promise<DiagnosisTaskEntity | null> {
    return this.taskStore.get(diagnosisId) ?? null;
  }

  async markRunning(diagnosisId: string): Promise<void> {
    const task = this.taskStore.get(diagnosisId);
    if (!task) return;

    task.status = 'running';
    task.updatedAt = new Date().toISOString();
    this.taskStore.set(diagnosisId, task);
  }

  async saveCompletedResult(diagnosisId: string, result: any): Promise<void> {
    const task = this.taskStore.get(diagnosisId);
    if (!task) return;

    task.status = 'completed';
    task.result = result;
    task.updatedAt = new Date().toISOString();
    this.taskStore.set(diagnosisId, task);
  }

  async markFailed(diagnosisId: string, errorMessage: string): Promise<void> {
    const task = this.taskStore.get(diagnosisId);
    if (!task) return;

    task.status = 'failed';
    task.errorMessage = errorMessage;
    task.updatedAt = new Date().toISOString();
    this.taskStore.set(diagnosisId, task);
  }

  async saveRawEvidence(
    diagnosisId: string,
    rawEvidence: RawEvidenceBundle,
  ): Promise<void> {
    this.evidenceStore.set(diagnosisId, rawEvidence);
  }

  async findRawEvidence(diagnosisId: string): Promise<RawEvidenceBundle | null> {
    return this.evidenceStore.get(diagnosisId) ?? null;
  }

  async findEvidence(
    diagnosisId: string,
    evidenceRef: string,
  ): Promise<any | null> {
    const raw = this.evidenceStore.get(diagnosisId);
    const evidences = raw?.evidences ?? [];
    return evidences.find((item) => item.evidence_ref === evidenceRef) ?? null;
  }
}
```

## 5.1 domain/rule-engine.service.ts

这里先内置 2 条规则骨架示例，后面你可以继续扩成规则清单。

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { RuleFinding } from '../models/rule-finding.model';

@Injectable()
export class RuleEngineService {
  run(context: DiagnosisContext): RuleFinding[] {
    const findings: RuleFinding[] = [];

    const r201 = this.hitFalsyValueSwallowed(context);
    if (r201) findings.push(r201);

    const r101 = this.hitApiFieldMissing(context);
    if (r101) findings.push(r101);

    return findings;
  }

  private hitFalsyValueSwallowed(
    context: DiagnosisContext,
  ): RuleFinding | null {
    const hasUpstreamZero =
      context.apiValue === 0 ||
      context.storeValue === 0 ||
      context.selectorValue === 0 ||
      context.renderInputValue === 0;

    const outputLooksFallback =
      context.formatterOutputIsFallback === true ||
      this.isFallbackValue(context.renderOutputValue, context.fallbackTokens);

    if (!hasUpstreamZero) return null;
    if (!outputLooksFallback) return null;

    return {
      ruleCode: 'R201',
      title: '合法 falsy 值被误判为空',
      diagnosisLabel: '合法值被判空吞掉',
      category: 'render_transform',
      severity: 'high',
      confidence: 0.94,
      layer: 'render',
      cluster: 'formatter_fallback',
      summary:
        '上游存在合法值 0，但格式化逻辑将其判定为空，最终输出兜底值。',
      evidenceRefs: ['ev_api_amount_0', 'ev_formatter_output_fallback'],
      suggestions: ['将 if (!value) 改为 value == null'],
      isSymptomOnly: false,
    };
  }

  private hitApiFieldMissing(context: DiagnosisContext): RuleFinding | null {
    if (context.responseSuccess !== true) return null;
    if (context.apiFieldExists !== false) return null;

    return {
      ruleCode: 'R101',
      title: '接口返回成功但字段缺失',
      diagnosisLabel: '接口字段缺失',
      category: 'data_source',
      severity: 'high',
      confidence: 0.92,
      layer: 'api',
      cluster: 'api_field_missing',
      summary: '接口请求成功，但目标字段未出现在返回 payload 中。',
      evidenceRefs: ['ev_api_field_missing'],
      suggestions: ['检查接口 schema 变更', '检查字段路径配置是否过期'],
      isSymptomOnly: false,
    };
  }

  private isFallbackValue(value: unknown, fallbackTokens?: string[]): boolean {
    if (typeof value !== 'string') return false;
    return (fallbackTokens ?? []).includes(value);
  }
}
```

## 6.1 domain/ranking.service.ts

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import {
  EnrichedFinding,
  RankScore,
  RuleFinding,
} from '../models/rule-finding.model';

@Injectable()
export class RankingService {
  rank(findings: RuleFinding[], context: DiagnosisContext): EnrichedFinding[] {
    return findings
      .map((finding) => {
        const score = this.calculateScore(finding, context);
        return {
          ...finding,
          rankScore: score,
          rankReasons: this.buildReasons(finding, score),
        };
      })
      .sort((a, b) => b.rankScore.finalScore - a.rankScore.finalScore);
  }

  private calculateScore(
    finding: RuleFinding,
    context: DiagnosisContext,
  ): RankScore {
    const baseScore = Math.round(finding.confidence * 100);

    const rootCauseScore = finding.isSymptomOnly ? 0 : 16;

    const specificityScore = finding.cluster ? 20 : 10;

    const evidenceScore = Math.min((finding.evidenceRefs?.length ?? 0) * 5, 20);

    const chainConsistencyScore =
      context.apiValue !== undefined &&
      context.renderOutputValue !== undefined
        ? 8
        : 2;

    const symptomPenalty = finding.isSymptomOnly ? 20 : 0;

    const duplicatePenalty = 0;

    const finalScore =
      baseScore +
      rootCauseScore +
      specificityScore +
      evidenceScore +
      chainConsistencyScore -
      symptomPenalty -
      duplicatePenalty;

    return {
      ruleCode: finding.ruleCode,
      baseScore,
      rootCauseScore,
      specificityScore,
      evidenceScore,
      chainConsistencyScore,
      symptomPenalty,
      duplicatePenalty,
      finalScore,
    };
  }

  private buildReasons(finding: RuleFinding, score: RankScore): string[] {
    const reasons: string[] = [];

    if (!finding.isSymptomOnly) reasons.push('根因层级优先');
    if (finding.cluster) reasons.push('规则具体性较高');
    if ((finding.evidenceRefs?.length ?? 0) >= 2) reasons.push('证据链较完整');
    if (score.baseScore >= 90) reasons.push('规则置信度高');

    return reasons;
  }
}
```

## 6.2 domain/diagnosis-conclusion.service.ts

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { DiagnosisConclusion } from '../models/diagnosis-conclusion.model';
import { RuleFinding } from '../models/rule-finding.model';
import { RankingService } from './ranking.service';

@Injectable()
export class DiagnosisConclusionService {
  constructor(private readonly rankingService: RankingService) {}

  conclude(
    findings: RuleFinding[],
    context: DiagnosisContext,
  ): DiagnosisConclusion {
    if (!findings.length) {
      return this.buildEmptyConclusion(context);
    }

    const ranked = this.rankingService.rank(findings, context);
    const topFinding = ranked[0] ?? null;

    const supportingFindings = ranked.filter(
      (item, index) =>
        index > 0 &&
        !item.isSymptomOnly &&
        item.cluster !== topFinding?.cluster,
    );

    const symptomFindings = ranked.filter((item) => item.isSymptomOnly);

    const repairHints = Array.from(
      new Set([
        ...(topFinding?.suggestions ?? []),
        ...supportingFindings.flatMap((item) => item.suggestions),
      ]),
    ).slice(0, 5);

    return {
      topFinding,
      supportingFindings,
      symptomFindings,
      summary: topFinding
        ? `最可能根因是「${topFinding.diagnosisLabel}」：${topFinding.summary}`
        : '当前未形成明确归因结论。',
      repairHints,
      scoreBreakdown: ranked.map((item) => ({
        ruleCode: item.ruleCode,
        finalScore: item.rankScore.finalScore,
        reason: item.rankReasons,
      })),
      diagnosisState:
        topFinding.confidence >= 0.9
          ? 'confirmed_root_cause'
          : 'probable_root_cause',
    };
  }

  private buildEmptyConclusion(
    context: DiagnosisContext,
  ): DiagnosisConclusion {
    const hasMinimalEvidence =
      Boolean(context.responseSuccess) ||
      Boolean(context.storeUpdated) ||
      Boolean(context.renderTriggered) ||
      context.domValue !== undefined;

    return {
      topFinding: null,
      supportingFindings: [],
      symptomFindings: [],
      summary: hasMinimalEvidence
        ? '当前链路存在证据，但现有规则未识别出明确异常。'
        : '当前证据不足，尚无法明确归因。',
      repairHints: hasMinimalEvidence
        ? ['检查业务计算逻辑', '补充更细的规则覆盖']
        : ['补充接口响应快照', '补充状态更新轨迹', '补充渲染证据'],
      scoreBreakdown: [],
      diagnosisState: hasMinimalEvidence
        ? 'no_rule_matched'
        : 'insufficient_evidence',
    };
  }
}
```

```ts
      label: `查看证据 ${ref}`,
```

```ts
    const brokenRuleCode = conclusion.topFinding?.ruleCode;

    const chain: Array<Record<string, any>> = [];

    if (context.apiFieldPath) {
      chain.push({
        step: chain.length + 1,
        layer: 'api',
        node_type: 'response_field',
        label: `${context.apiFieldPath} = ${String(context.apiValue)}`,
        status: context.responseSuccess ? 'ok' : 'observed',
        evidence_ref: 'ev_api_amount_0',
      });
    }

    if (context.storeKey) {
      chain.push({
        step: chain.length + 1,
        layer: 'state',
        node_type: 'store_update',
        label: `${context.storeKey} = ${String(context.storeValue)}`,
        status: context.storeUpdated ? 'ok' : 'affected',
        evidence_ref: 'ev_store_value',
      });
    }

    if (context.formatterName) {
      chain.push({
        step: chain.length + 1,
        layer: 'render',
        node_type: 'formatter',
        label: `${context.formatterName}(${String(context.renderInputValue)}) -> ${String(context.renderOutputValue)}`,
        status: brokenRuleCode === 'R201' ? 'broken' : 'ok',
        evidence_ref: 'ev_formatter_output_fallback',
        hit_rule_code: brokenRuleCode === 'R201' ? brokenRuleCode : undefined,
      });
    }

    chain.push({
      step: chain.length + 1,
      layer: 'dom',
      node_type: 'dom_text',
      label: `DOM displays ${String(context.domValue ?? context.displayedValue)}`,
      status: conclusion.topFinding ? 'affected' : 'observed',
      evidence_ref: 'ev_dom_snapshot',
    });

    return chain;
  }
}
```

## 7.2 application/diagnosis-query.service.ts

这是最核心的主流程代码骨架。

```ts
import { Injectable, NotFoundException } from '@nestjs/common';
import { GetDiagnosisResponseDto } from '../dto/diagnosis-result.dto';
import { GetEvidenceResponseDto } from '../dto/evidence-detail.dto';
import { GetDominoChainResponseDto } from '../dto/domino-chain.dto';
import { DiagnosisConclusionService } from '../domain/diagnosis-conclusion.service';
import { DominoChainBuilder } from '../domain/domino-chain.builder';
import { ExplanationBuilder } from '../domain/explanation.builder';
import { RuleEngineService } from '../domain/rule-engine.service';
import { ContextBuilder } from '../infrastructure/context.builder';
import { DiagnosisRepository } from '../infrastructure/diagnosis.repository';
import { EvidenceCollector } from '../infrastructure/evidence.collector';

@Injectable()
export class DiagnosisQueryService {
  constructor(
    private readonly diagnosisRepository: DiagnosisRepository,
    private readonly evidenceCollector: EvidenceCollector,
    private readonly contextBuilder: ContextBuilder,
    private readonly ruleEngineService: RuleEngineService,
    private readonly diagnosisConclusionService: DiagnosisConclusionService,
    private readonly explanationBuilder: ExplanationBuilder,
    private readonly dominoChainBuilder: DominoChainBuilder,
  ) {}

  async getOrRunDiagnosis(
    diagnosisId: string,
  ): Promise<GetDiagnosisResponseDto> {
    const task = await this.diagnosisRepository.findById(diagnosisId);
    if (!task) {
      throw new NotFoundException(`Diagnosis ${diagnosisId} not found`);
    }

    if (task.status === 'completed' && task.result) {
      return {
        code: 'OK',
        message: 'success',
        data: task.result,
      };
    }

    await this.diagnosisRepository.markRunning(diagnosisId);

    try {
      const rawEvidence = await this.evidenceCollector.collect(task);
      await this.diagnosisRepository.saveRawEvidence(diagnosisId, rawEvidence);

      const context = await this.contextBuilder.build(task, rawEvidence);
      const findings = this.ruleEngineService.run(context);
      const conclusion = this.diagnosisConclusionService.conclude(
        findings,
        context,
      );
      const explanation = this.explanationBuilder.build(context, conclusion);
      const dominoChain = this.dominoChainBuilder.build(context, conclusion);

      const result = {
        diagnosis_id: diagnosisId,
        mode: task.mode,
        status: 'completed',
        diagnosis_state: conclusion.diagnosisState,
        interaction_id: context.interactionId ?? null,
        lineage_id: context.lineageId ?? null,
        target: explanation.target,
        top_cause: explanation.topCause,
        supporting_causes: explanation.supportingCauses,
        symptoms: explanation.symptoms,
        summary: explanation.summary,
        repair_hints: explanation.repairHints,
        score_breakdown:
          task.options?.include_score_breakdown === false
            ? undefined
            : explanation.scoreBreakdown,
        evidence_overview:
          task.options?.include_evidence_overview === false
            ? undefined
            : explanation.evidenceOverview,
        domino_chain:
          task.options?.include_domino_chain === false
            ? undefined
            : dominoChain,
        next_actions: explanation.nextActions,
        generated_at: new Date().toISOString(),
      };

      await this.diagnosisRepository.saveCompletedResult(diagnosisId, result);

      return {
        code: 'OK',
        message: 'success',
        data: result,
      };
    } catch (error) {
      await this.diagnosisRepository.markFailed(
        diagnosisId,
        error instanceof Error ? error.message : 'unknown error',
      );
      throw error;
    }
  }

  async getEvidenceDetail(
    diagnosisId: string,
    evidenceRef: string,
  ): Promise<GetEvidenceResponseDto> {
    const evidence = await this.diagnosisRepository.findEvidence(
      diagnosisId,
      evidenceRef,
    );

    if (!evidence) {
      throw new NotFoundException(`Evidence ${evidenceRef} not found`);
    }

    return {
      code: 'OK',
      message: 'success',
      data: evidence,
    };
  }

  async getDominoChain(
    diagnosisId: string,
  ): Promise<GetDominoChainResponseDto> {
    const task = await this.diagnosisRepository.findById(diagnosisId);

    if (!task || !task.result) {
      throw new NotFoundException(`Diagnosis ${diagnosisId} not found`);
    }

    return {
      code: 'OK',
      message: 'success',
      data: {
        diagnosis_id: diagnosisId,
        domino_chain: task.result.domino_chain ?? [],
      },
    };
  }
}
```
