下面直接给你 **DiagnosisModule 真正模块文件骨架**。  
目标是把你前面已经定下来的链路真正落成一个可接入 NestJS 的模块：

- `Controller`
- `Application Services`
- `Context Builder / Evidence Collector`
- `Rule Engine`
- `Ranking`
- `Conclusion`
- `Explanation`
- `Domino Chain`
- `Repository token/provider`
- `Module wiring`

我会按你当前已经确认的 MVP 策略来给：

- `POST /api/v1/diagnosis` 只创建任务
- `GET /api/v1/diagnosis/:id` lazy execution
- repository 用 token
- service / builder 用 class provider
- 先给 **可落地骨架**
- 某些 domain model 仍允许 `any/unknown` 简化 Swagger/DTO 波动

## 10.2 `diagnosis-query.service.ts`

```ts
import { Inject, Injectable, NotFoundException } from '@nestjs/common';

import { DiagnosisConclusionService } from '../../domain/services/diagnosis-conclusion.service';
import { DiagnosisContextBuilderPort } from '../../domain/interfaces/diagnosis-context-builder.port';
import { DiagnosisRepository } from '../../domain/interfaces/diagnosis-repository.interface';
import { DiagnosisResult } from '../../domain/models/diagnosis-result.model';
import { DominoChainBuilder } from '../../domain/builders/domino-chain.builder';
import { ExplanationBuilder } from '../../domain/builders/explanation.builder';
import { RuleEngineService } from '../../domain/services/rule-engine.service';
import { DIAGNOSIS_REPOSITORY } from '../../infrastructure/constants/diagnosis-repository.token';

@Injectable()
export class DiagnosisQueryService {
  constructor(
    @Inject(DIAGNOSIS_REPOSITORY)
    private readonly repository: DiagnosisRepository,
    @Inject('DiagnosisContextBuilder')
    private readonly contextBuilder: DiagnosisContextBuilderPort,
    private readonly ruleEngineService: RuleEngineService,
    private readonly diagnosisConclusionService: DiagnosisConclusionService,
    private readonly explanationBuilder: ExplanationBuilder,
    private readonly dominoChainBuilder: DominoChainBuilder,
  ) {}

  async getById(id: string): Promise<DiagnosisResult> {
    const task = await this.repository.findTaskById(id);
    if (!task) {
      throw new NotFoundException('Diagnosis task not found');
    }

    const cached = await this.repository.findResultByDiagnosisId(id);
    if (cached) {
      return cached;
    }

    const context = await this.contextBuilder.build(task);
    const findings = this.ruleEngineService.run(context);
    const conclusion = this.diagnosisConclusionService.conclude(
      findings,
      context,
    );
    const explanation = this.explanationBuilder.build(conclusion, context);
    const dominoChain = this.dominoChainBuilder.build(conclusion, context);

    const result: DiagnosisResult = {
      diagnosisId: id,
      status: 'completed',
      createdAt: task.createdAt,
      updatedAt: new Date().toISOString(),
      context,
      findings,
      conclusion,
      explanation,
      dominoChain,
    };

    await this.repository.saveResult(result);

    return result;
  }
}
```

## 12.1 `rule-engine.service.ts`

```ts
import { Injectable } from '@nestjs/common';

import { DiagnosisContext } from '../models/diagnosis-context.model';
import { RuleFinding } from '../models/rule-finding.model';
import { DiagnosisRuleRegistry } from '../rules/registry/diagnosis-rule.registry';

@Injectable()
export class RuleEngineService {
  constructor(private readonly registry: DiagnosisRuleRegistry) {}

  run(context: DiagnosisContext): RuleFinding[] {
    const findings = this.registry
      .getAll()
      .map((rule) => rule.evaluate(context))
      .filter((item): item is RuleFinding => Boolean(item));

    const deduped = new Map<string, RuleFinding>();

    for (const finding of findings) {
      const key = `${finding.ruleCode}:${finding.layer}:${finding.cluster}`;
      if (!deduped.has(key)) {
        deduped.set(key, finding);
      }
    }

    return Array.from(deduped.values());
  }
}
```

## 12.2 `ranking.service.ts`

```ts
import { Injectable } from '@nestjs/common';

import { DiagnosisContext } from '../models/diagnosis-context.model';
import { RankedFinding } from '../models/ranked-finding.model';
import { RuleFinding } from '../models/rule-finding.model';

@Injectable()
export class RankingService {
  private readonly layerWeight: Record<string, number> = {
    interaction: 0.1,
    handler: 0.095,
    request: 0.09,
    api: 0.085,
    state: 0.06,
    selector: 0.05,
    render: 0.04,
    dom: 0.03,
    ui: 0.01,
    request_to_ui: 0.02,
  };

  rank(findings: RuleFinding[], _context: DiagnosisContext): RankedFinding[] {
    const clusterLeader = new Map<string, number>();

    for (const finding of findings) {
      const prev = clusterLeader.get(finding.cluster) ?? -1;
      if (finding.confidence > prev) {
        clusterLeader.set(finding.cluster, finding.confidence);
      }
    }

    return findings
      .map((finding): RankedFinding => {
        const baseConfidence = finding.confidence;
        const rootCauseBonus = finding.isSymptomOnly ? 0 : 0.2;
        const symptomPenalty = finding.isSymptomOnly ? 0.25 : 0;
        const upstreamBonus = this.layerWeight[finding.layer] ?? 0;
        const duplicatePenalty =
          clusterLeader.get(finding.cluster) === finding.confidence ? 0 : 0.12;
        const evidenceScore = Math.min(finding.evidenceRefs.length * 0.03, 0.09);
        const clusterBonus = 0.03;

        const finalScore =
          baseConfidence +
          rootCauseBonus +
          upstreamBonus +
          evidenceScore +
          clusterBonus -
          symptomPenalty -
          duplicatePenalty;

        const rankReasons: string[] = [];
        if (!finding.isSymptomOnly) rankReasons.push('根因规则优先');
        if (finding.isSymptomOnly) rankReasons.push('症状规则已降权');
        if (upstreamBonus > 0.07) rankReasons.push('上游链路优先级更高');
        if (duplicatePenalty > 0) rankReasons.push('同 cluster 规则已做去重降权');

        return {
          ...finding,
          rankScore: {
            baseConfidence,
            rootCauseBonus,
            symptomPenalty,
            upstreamBonus,
            duplicatePenalty,
            evidenceScore,
            clusterBonus,
            finalScore,
          },
          rankReasons,
        };
      })
      .sort((a, b) => b.rankScore.finalScore - a.rankScore.finalScore);
  }
}
```

## 12.3 `diagnosis-conclusion.service.ts`

```ts
import { Injectable } from '@nestjs/common';

import { DiagnosisState } from '../enums/diagnosis-state.enum';
import { DiagnosisConclusion } from '../models/diagnosis-conclusion.model';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { RuleFinding } from '../models/rule-finding.model';
import { RankingService } from './ranking.service';

@Injectable()
export class DiagnosisConclusionService {
  constructor(private readonly rankingService: RankingService) {}

  conclude(
    findings: RuleFinding[],
    context: DiagnosisContext,
  ): DiagnosisConclusion {
    const ranked = this.rankingService.rank(findings, context);

    if (ranked.length === 0) {
      const weakEvidence =
        !context.responseSuccess &&
        !context.storeUpdated &&
        !context.renderTriggered;

      return {
        topFinding: null,
        supportingFindings: [],
        symptomFindings: [],
        diagnosisState: weakEvidence
          ? DiagnosisState.INSUFFICIENT_EVIDENCE
          : DiagnosisState.NO_RULE_MATCHED,
        summary: weakEvidence
          ? '当前证据不足，无法确认明确根因。'
          : '现有规则未识别出明确异常。',
        repairHints: [],
        scoreBreakdown: [],
      };
    }

    const topFinding = ranked.find((item) => !item.isSymptomOnly) ?? ranked[0];
    const symptomFindings = ranked.filter((item) => item.isSymptomOnly);
    const supportingFindings = ranked.filter(
      (item) =>
        !item.isSymptomOnly &&
        item.ruleCode !== topFinding.ruleCode &&
        item.cluster !== topFinding.cluster,
    );

    const repairHints = Array.from(
      new Set([
        ...topFinding.suggestions,
        ...supportingFindings.flatMap((item) => item.suggestions),
      ]),
    );

    return {
      topFinding,
      supportingFindings,
      symptomFindings,
      diagnosisState:
        topFinding.confidence >= 0.9
          ? DiagnosisState.CONFIRMED_ROOT_CAUSE
          : DiagnosisState.PROBABLE_ROOT_CAUSE,
      summary:
        topFinding.confidence >= 0.9
          ? `已定位到根因：${topFinding.diagnosisLabel}。`
          : `高概率根因：${topFinding.diagnosisLabel}。`,
      repairHints,
      scoreBreakdown: ranked.map((item) => ({
        ruleCode: item.ruleCode,
        finalScore: item.rankScore.finalScore,
        reason: item.rankReasons,
      })),
    };
  }
}
```

## 13.1 `explanation.builder.ts`

```ts
import { Injectable } from '@nestjs/common';

import { DiagnosisConclusion } from '../models/diagnosis-conclusion.model';
import {
  DiagnosisContext,
} from '../models/diagnosis-context.model';
import { DiagnosisExplanation } from '../models/diagnosis-result.model';

@Injectable()
export class ExplanationBuilder {
  build(
    conclusion: DiagnosisConclusion,
    _context: DiagnosisContext,
  ): DiagnosisExplanation {
    if (!conclusion.topFinding) {
      return {
        summaryText: '当前证据不足，暂时无法确认明确根因。',
        evidenceNarrative: [],
        operatorAdvice: conclusion.repairHints.map(
          (item) => `建议处理：${item}`,
        ),
        symptomNotes: [],
      };
    }

    return {
      summaryText: `${conclusion.summary}${conclusion.topFinding.summary}`,
      evidenceNarrative: [
        `核心证据：${conclusion.topFinding.diagnosisLabel}。${conclusion.topFinding.summary}。证据引用: ${conclusion.topFinding.evidenceRefs.join(', ')}。`,
        ...conclusion.supportingFindings.map(
          (item) =>
            `辅助证据：${item.diagnosisLabel}。${item.summary}。证据引用: ${item.evidenceRefs.join(', ')}。`,
        ),
      ],
      operatorAdvice: conclusion.repairHints.map(
```

```ts
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { DominoChain, DominoNode } from '../models/diagnosis-result.model';

@Injectable()
export class DominoChainBuilder {
  build(
    conclusion: DiagnosisConclusion,
    context: DiagnosisContext,
  ): DominoChain {
    const nodes =
      context.mode === 'click_diagnosis'
        ? this.buildClickNodes()
        : this.buildInspectNodes();

    const mark = (
      layer: string,
      patch: Partial<DominoNode>,
      relatedRuleCode: string,
      evidenceRefs: string[],
    ) => {
      const node = nodes.find((item) => item.layer === layer || item.type === layer);
      if (!node) return;

      node.status = patch.status ?? node.status;
      node.isRootCause = patch.isRootCause ?? node.isRootCause;
      node.isSupporting = patch.isSupporting ?? node.isSupporting;
      node.isSymptom = patch.isSymptom ?? node.isSymptom;
      node.relatedRuleCodes = Array.from(
        new Set([...node.relatedRuleCodes, relatedRuleCode]),
      );
      node.evidenceRefs = Array.from(new Set([...node.evidenceRefs, ...evidenceRefs]));
    };

    if (conclusion.topFinding) {
      mark(
        conclusion.topFinding.layer,
        { status: 'broken', isRootCause: true },
        conclusion.topFinding.ruleCode,
        conclusion.topFinding.evidenceRefs,
      );
    }

    for (const item of conclusion.supportingFindings) {
      mark(
        item.layer,
        { status: 'suspect', isSupporting: true },
        item.ruleCode,
        item.evidenceRefs,
      );
    }

      mark(
        item.layer,
        { status: 'symptom', isSymptom: true },
        item.ruleCode,
        item.evidenceRefs,
      );
```

```ts
import { Injectable } from '@nestjs/common';

import { DiagnosisRepository } from '../../domain/interfaces/diagnosis-repository.interface';

@Injectable()
export class InMemoryDiagnosisRepository implements DiagnosisRepository {
  private readonly tasks = new Map<string, DiagnosisTask>();
  private readonly results = new Map<string, DiagnosisResult>();

  async createTask(task: DiagnosisTask): Promise<void> {
    this.tasks.set(task.id, task);
  }

  async findResultByDiagnosisId(id: string): Promise<DiagnosisResult | null> {
    return this.results.get(id) ?? null;
  }
```
