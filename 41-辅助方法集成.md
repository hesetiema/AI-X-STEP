下面直接给你 **“把 helper 接入已有 spec 的重构示例版”**。  
我挑三类最有代表性的文件来改：

1. `ranking.service.spec.ts`
2. `diagnosis-conclusion.service.spec.ts`
3. `integration/diagnosis-query.service.integration.spec.ts`

这样你后面可以按同样模式改其它 spec。

# 1. 先约定 helper 引用路径

假设你已经有：

```bash
src/modules/diagnosis/__tests__/helpers/
├── build-context.ts
├── build-finding.ts
├── build-ranked-finding.ts
├── build-conclusion.ts
├── build-task.ts
├── build-result.ts
├── build-domino-chain.ts
├── build-explanation.ts
└── index.ts
```

那么在 `__tests__` 下的 spec 里：

```ts
import { ... } from './helpers';
```

在 `__tests__/integration` 下的 spec 里：

```ts
import { ... } from '../helpers';
```

## 重构后版本

### `src/modules/diagnosis/__tests__/ranking.service.spec.ts`

```ts
import { RankingService } from '../domain/services/ranking.service';
import {
  buildDiagnosisContext,
  buildRuleFinding,
} from './helpers';

describe('RankingService', () => {
  let service: RankingService;

  beforeEach(() => {
    service = new RankingService();
  });

  it('should rank root-cause finding above symptom finding when confidence is equal', () => {
    const rootCause = buildRuleFinding({
      ruleCode: 'R201',
      confidence: 0.8,
      layer: 'render',
      isSymptomOnly: false,
      cluster: 'root_cluster',
    });

    const symptom = buildRuleFinding({
      ruleCode: 'R501',
      confidence: 0.8,
      layer: 'ui',
      isSymptomOnly: true,
      cluster: 'symptom_cluster',
    });

    const result = service.rank(
      [symptom, rootCause],
      buildDiagnosisContext(),
    );

    expect(result[0].ruleCode).toBe('R201');
    expect(result[1].ruleCode).toBe('R501');
    expect(result[0].rankScore.finalScore).toBeGreaterThan(
      result[1].rankScore.finalScore,
    );
  });

  it('should rank upstream api finding above downstream dom finding when confidence is equal', () => {
    const apiFinding = buildRuleFinding({
      ruleCode: 'R101',
      confidence: 0.85,
      layer: 'api',
      cluster: 'api_cluster',
      isSymptomOnly: false,
    });

    const domFinding = buildRuleFinding({
      ruleCode: 'R302',
      confidence: 0.85,
      layer: 'dom',
      cluster: 'dom_cluster',
      isSymptomOnly: false,
    });

    const result = service.rank(
      [domFinding, apiFinding],
      buildDiagnosisContext(),
    );

    expect(result[0].ruleCode).toBe('R101');
    expect(result[1].ruleCode).toBe('R302');
  });

  it('should apply duplicate penalty to non-leader finding in the same cluster', () => {
    const clusterLeader = buildRuleFinding({
      ruleCode: 'R201',
      confidence: 0.95,
      layer: 'render',
      cluster: 'formatter_cluster',
      isSymptomOnly: false,
    });

    const duplicate = buildRuleFinding({
      ruleCode: 'R202',
      confidence: 0.83,
      layer: 'render',
      cluster: 'formatter_cluster',
      isSymptomOnly: false,
    });

    const result = service.rank(
      [duplicate, clusterLeader],
      buildDiagnosisContext(),
    );

    const leader = result.find((item) => item.ruleCode === 'R201');
    const dup = result.find((item) => item.ruleCode === 'R202');

    expect(leader).toBeDefined();
    expect(dup).toBeDefined();
    expect(dup!.rankScore.duplicatePenalty).toBeGreaterThan(0);
    expect(leader!.rankScore.duplicatePenalty).toBe(0);
    expect(leader!.rankScore.finalScore).toBeGreaterThan(
      dup!.rankScore.finalScore,
    );
  });

  it('should give higher score to finding with richer evidence refs', () => {
    const lowEvidence = buildRuleFinding({
      ruleCode: 'R103',
      confidence: 0.8,
      layer: 'state',
      cluster: 'state_cluster_a',
      evidenceRefs: ['ev1'],
    });

    const highEvidence = buildRuleFinding({
      ruleCode: 'R104',
      confidence: 0.8,
      layer: 'state',
      cluster: 'state_cluster_b',
      evidenceRefs: ['ev1', 'ev2', 'ev3'],
    });

    const result = service.rank(
      [lowEvidence, highEvidence],
      buildDiagnosisContext(),
    );

    const low = result.find((item) => item.ruleCode === 'R103');
    const high = result.find((item) => item.ruleCode === 'R104');

    expect(high).toBeDefined();
    expect(low).toBeDefined();
    expect(high!.rankScore.evidenceScore).toBeGreaterThan(
      low!.rankScore.evidenceScore,
    );
    expect(high!.rankScore.finalScore).toBeGreaterThan(
      low!.rankScore.finalScore,
    );
  });

  it('should keep both findings but lower score for symptom rule', () => {
    const rootCause = buildRuleFinding({
      ruleCode: 'R302',
      confidence: 0.76,
      layer: 'dom',
      cluster: 'dom_cluster_a',
      isSymptomOnly: false,
    });

    const symptom = buildRuleFinding({
      ruleCode: 'R502',
      confidence: 0.9,
      layer: 'dom',
      cluster: 'dom_cluster_b',
      isSymptomOnly: true,
    });

    const result = service.rank(
      [symptom, rootCause],
      buildDiagnosisContext(),
    );

    expect(result).toHaveLength(2);
    expect(result[0].ruleCode).toBe('R302');
    expect(result[1].ruleCode).toBe('R502');
  });

  it('should sort findings by final score descending', () => {
    const findings = [
      buildRuleFinding({
        ruleCode: 'R501',
        confidence: 0.8,
        layer: 'ui',
        cluster: 'symptom_a',
        isSymptomOnly: true,
      }),
      buildRuleFinding({
        ruleCode: 'R101',
        confidence: 0.88,
        layer: 'api',
        cluster: 'api_a',
        isSymptomOnly: false,
        evidenceRefs: ['ev1', 'ev2'],
      }),
      buildRuleFinding({
        ruleCode: 'R302',
        confidence: 0.84,
        layer: 'dom',
        cluster: 'dom_a',
        isSymptomOnly: false,
      }),
    ];

    const result = service.rank(findings, buildDiagnosisContext());

    expect(result.map((item) => item.ruleCode)).toEqual([
      'R101',
      'R302',
      'R501',
    ]);
  });
});
```

```ts
import { DiagnosisConclusionService } from '../domain/services/diagnosis-conclusion.service';
import { RankingService } from '../domain/services/ranking.service';
import { DiagnosisState } from '../domain/enums/diagnosis-state.enum';
import {
  buildDiagnosisContext,
  buildRankedFinding,
} from './helpers';

describe('DiagnosisConclusionService', () => {
  let service: DiagnosisConclusionService;
  let rankingService: jest.Mocked<Pick<RankingService, 'rank'>>;

  beforeEach(() => {
    rankingService = {
      rank: jest.fn(),
    };

    service = new DiagnosisConclusionService(
      rankingService as unknown as RankingService,
    );
  });

  it('should call ranking service with findings and context', () => {
    const findings = [
      buildRankedFinding({
        ruleCode: 'R101',
      }),
    ];
    const context = buildDiagnosisContext();

    rankingService.rank.mockReturnValue([]);

    service.conclude(findings, context);

    expect(rankingService.rank).toHaveBeenCalledTimes(1);
    expect(rankingService.rank).toHaveBeenCalledWith(findings, context);
  });

  it('should select first ranked root-cause finding as topFinding', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R101',
        diagnosisLabel: '接口字段缺失',
        cluster: 'api_field_missing',
        confidence: 0.95,
      }),
      buildRankedFinding({
        ruleCode: 'R202',
        diagnosisLabel: 'formatter 输出占位值',
        cluster: 'formatter_cluster',
        confidence: 0.88,
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.topFinding?.ruleCode).toBe('R101');
    expect(result.topFinding?.diagnosisLabel).toBe('接口字段缺失');
  });

  it('should collect supportingFindings excluding topFinding itself and same-cluster root causes', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R201',
        cluster: 'formatter_cluster',
        diagnosisLabel: '合法值被判空吞掉',
      }),
      buildRankedFinding({
        ruleCode: 'R202',
        cluster: 'formatter_cluster',
        diagnosisLabel: 'formatter 输出占位值',
      }),
      buildRankedFinding({
        ruleCode: 'R103',
        cluster: 'state_cluster',
        diagnosisLabel: '请求成功但状态未更新',
      }),
      buildRankedFinding({
        ruleCode: 'R301',
        cluster: 'dom_cluster',
        diagnosisLabel: 'DOM 被隐藏',
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.topFinding?.ruleCode).toBe('R201');
    expect(result.supportingFindings.map((item) => item.ruleCode)).toEqual([
      'R103',
      'R301',
    ]);
    expect(result.supportingFindings.map((item) => item.ruleCode)).not.toContain(
      'R202',
    );
  });

  it('should collect symptomFindings separately', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R201',
        isSymptomOnly: false,
        cluster: 'root_cluster',
      }),
      buildRankedFinding({
        ruleCode: 'R501',
        isSymptomOnly: true,
        cluster: 'symptom_cluster_a',
      }),
      buildRankedFinding({
        ruleCode: 'R502',
        isSymptomOnly: true,
        cluster: 'symptom_cluster_b',
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.symptomFindings.map((item) => item.ruleCode)).toEqual([
      'R501',
      'R502',
    ]);
  });

  it('should set diagnosisState to confirmed_root_cause when top finding confidence >= 0.9', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R101',
        confidence: 0.95,
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.diagnosisState).toBe(DiagnosisState.CONFIRMED_ROOT_CAUSE);
  });

  it('should set diagnosisState to probable_root_cause when top finding confidence < 0.9', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R302',
        confidence: 0.84,
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.diagnosisState).toBe(DiagnosisState.PROBABLE_ROOT_CAUSE);
  });

  it('should set diagnosisState to insufficient_evidence when no ranked finding exists and upstream evidence is weak', () => {
    rankingService.rank.mockReturnValue([]);

    const result = service.conclude(
      [],
      buildDiagnosisContext({
        responseSuccess: false,
        storeUpdated: false,
        renderTriggered: false,
      }),
    );

    expect(result.diagnosisState).toBe(DiagnosisState.INSUFFICIENT_EVIDENCE);
    expect(result.summary).toBe('当前证据不足，无法确认明确根因。');
  });

  it('should set diagnosisState to no_rule_matched when no ranked finding exists but evidence is not weak', () => {
    rankingService.rank.mockReturnValue([]);

    const result = service.conclude(
      [],
      buildDiagnosisContext({
        responseSuccess: true,
        storeUpdated: false,
        renderTriggered: false,
      }),
    );

    expect(result.diagnosisState).toBe(DiagnosisState.NO_RULE_MATCHED);
    expect(result.summary).toBe('现有规则未识别出明确异常。');
  });

  it('should merge repair hints from topFinding and supportingFindings with deduplication', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R201',
        cluster: 'root_cluster',
        suggestions: ['检查 formatter 判空逻辑', '保留 0 值'],
      }),
      buildRankedFinding({
        ruleCode: 'R103',
        cluster: 'state_cluster',
        suggestions: ['检查 dispatch/commit 是否执行', '保留 0 值'],
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.repairHints).toEqual([
      '检查 formatter 判空逻辑',
      '保留 0 值',
      '检查 dispatch/commit 是否执行',
    ]);
  });

  it('should generate scoreBreakdown from ranked findings', () => {
    rankingService.rank.mockReturnValue([
      buildRankedFinding({
        ruleCode: 'R101',
        rankScore: {
          baseConfidence: 0.95,
          rootCauseBonus: 0.2,
          symptomPenalty: 0,
          upstreamBonus: 0.085,
          duplicatePenalty: 0,
          evidenceScore: 0.09,
          clusterBonus: 0.03,
          finalScore: 1.355,
        },
        rankReasons: ['根因规则优先', '上游 API 层优先级最高'],
      }),
      buildRankedFinding({
        ruleCode: 'R501',
        isSymptomOnly: true,
        rankScore: {
          baseConfidence: 0.8,
          rootCauseBonus: 0,
          symptomPenalty: 0.25,
          upstreamBonus: 0.03,
          duplicatePenalty: 0,
          evidenceScore: 0.03,
          clusterBonus: 0.03,
          finalScore: 0.61,
        },
        rankReasons: ['症状规则已降权'],
      }),
    ]);

    const result = service.conclude([], buildDiagnosisContext());

    expect(result.scoreBreakdown).toEqual([
      {
        ruleCode: 'R101',
        finalScore: 1.355,
        reason: ['根因规则优先', '上游 API 层优先级最高'],
      },
      {
        ruleCode: 'R501',
        finalScore: 0.61,
        reason: ['症状规则已降权'],
      },
    ]);
  });
});
```

# 4. 一个小纠正建议

上面第一条测试如果只是为了验证：

```ts
rankingService.rank(findings, context)
```

那 `findings` 更适合用 `buildRuleFinding()`，不是 `buildRankedFinding()`。  
因为 `conclude()` 的入参一般是 `RuleFinding[]`，不是 `RankedFinding[]`。

所以更严谨一点，你可以改成：

```ts
import {
  buildDiagnosisContext,
  buildRuleFinding,
  buildRankedFinding,
} from './helpers';
```

然后：

```ts
const findings = [
  buildRuleFinding({
    ruleCode: 'R101',
  }),
];
```

这个更准确。

```ts
      storeUpdated: false,
      renderTriggered: false,
    });

    const findings: ReturnType<typeof buildRuleFinding>[] = [];

    const conclusion = buildDiagnosisConclusion({
      topFinding: null,
      supportingFindings: [],
      symptomFindings: [],
      diagnosisState: 'insufficient_evidence' as any,
      summary: '当前证据不足，无法确认明确根因。',

    const explanation = buildDiagnosisExplanation({
```

# 6. 重构后的收益总结

接入 helper 后，这三类 spec 会明显改善：

---

## 6.1 默认值统一
例如所有测试 context 都默认：

```ts
responseSuccess: true
storeUpdated: true
renderTriggered: true
```

不会某些 spec 默认健康、某些默认缺字段。

---

## 6.2 spec 关注点更聚焦
以前测试里 20 行 builder 噪音，现在只看：

```ts
buildRuleFinding({ ruleCode: 'R101', layer: 'api' })
```

更容易看出测试意图。

---

## 6.3 后面字段扩展成本低
比如你后面给 `DiagnosisContext` 新增：

```ts
traceId?: string
```

只要改一处 helper。

# 7. 我建议你顺手再做一个小升级

现在的 helper 是 **基础构造器**。  
接下来最值得补的是 **场景级 fixture**，比如：

---

## inspect 场景
```ts
buildInspectApiFieldMissingScenario()
buildInspectFalsySwallowedScenario()
buildInspectDomNotUpdatedScenario()
```

## click 场景
```ts
buildClickHandlerNotStartedScenario()
buildClickRequestNotSentScenario()
```

这些 fixture 可以直接返回：

```ts
{
  task,
  context,
  findings,
  conclusion,
  explanation,
  dominoChain,
}
```

这样 integration spec 会更短。
