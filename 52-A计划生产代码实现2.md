```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisTaskRepository } from '../../application/ports/diagnosis-task.repository';
import {
  DiagnosisResult,
  DiagnosisTaskEntity,
} from '../../domain/types/diagnosis.types';

@Injectable()
export class InMemoryDiagnosisTaskRepository implements DiagnosisTaskRepository {
  private readonly store = new Map<string, DiagnosisTaskEntity>();
  private sequence = 0;

  async create(input: Record<string, unknown>): Promise<DiagnosisTaskEntity> {
    this.sequence += 1;
    const now = new Date().toISOString();

    const entity: DiagnosisTaskEntity = {
      id: `diag-${this.sequence}`,
      status: 'pending',
      input,
      createdAt: now,
      updatedAt: now,
    };

    this.store.set(entity.id, entity);
    return entity;
  }

  async findById(id: string): Promise<DiagnosisTaskEntity | null> {
    return this.store.get(id) ?? null;
  }

  async saveResult(
    id: string,
    result: DiagnosisResult,
  ): Promise<DiagnosisTaskEntity> {
    const entity = this.store.get(id);

    if (!entity) {
      throw new Error(`Diagnosis task not found: ${id}`);
    }

    const updated: DiagnosisTaskEntity = {
      ...entity,
      status: 'completed',
      result,
      updatedAt: new Date().toISOString(),
    };

    this.store.set(id, updated);
    return updated;
  }
}
```

```ts
import { Injectable } from '@nestjs/common';
import {
  DiagnosisContext,
  DiagnosisEvidenceItem,
  DiagnosisTaskEntity,
} from '../../domain/types/diagnosis.types';

@Injectable()
export class DiagnosisContextBuilder {
  build(task: DiagnosisTaskEntity): DiagnosisContext {
    const input = task.input ?? {};

    return {
      traceId: String(input.traceId ?? ''),
      interactionId: String(input.interactionId ?? `interaction-${task.id}`),
      targetId: String(input.targetId ?? ''),
      sessionId: input.sessionId ? String(input.sessionId) : undefined,
      occurredAt: String(input.occurredAt ?? task.createdAt),
      symptoms: Array.isArray(input.symptoms)
        ? input.symptoms.map((x) => String(x))
        : [],
      evidence: this.normalizeEvidence(input.evidence),
      meta:
        input.meta && typeof input.meta === 'object'
          ? (input.meta as Record<string, unknown>)
          : {},
    };
  }

  private normalizeEvidence(raw: unknown): DiagnosisEvidenceItem[] {
    if (!Array.isArray(raw)) {
      return [];
    }

    return raw.map((item, index) => {
      const value = item as Record<string, unknown>;
      return {
        id: String(value?.id ?? `ev-${index + 1}`),
        type: String(value?.type ?? 'unknown'),
        label: String(value?.label ?? `evidence-${index + 1}`),
        value: value?.value,
        source: value?.source ? String(value.source) : undefined,
      };
    });
  }
}
```

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisContext, RuleFinding } from '../types/diagnosis.types';
import { DiagnosisRuleRegistry } from './rule-registry';

@Injectable()
export class RuleEngineService {
  constructor(
    private readonly ruleRegistry: DiagnosisRuleRegistry,
  ) {}

  evaluate(context: DiagnosisContext): RuleFinding[] {
    const rules = this.ruleRegistry.getRules();
    const allFindings = rules.flatMap((rule) => rule.evaluate(context));

    const deduped = new Map<string, RuleFinding>();

    for (const finding of allFindings) {
      const key = `${finding.ruleCode}:${finding.layer}:${finding.cluster}`;

      const existing = deduped.get(key);
      if (!existing || finding.confidence > existing.confidence) {
        deduped.set(key, finding);
      }
    }

    return [...deduped.values()];
  }
}
```

# 八、RankingService

> 这里按你已确认的策略实现：
- root cause 优先于 symptom
- 上游层优先
- 同 cluster 次条目降权
- evidence 多者略加分
- 最终按 `finalScore` 排序

---

## `src/diagnosis/application/services/ranking.service.ts`

```ts
import { Injectable } from '@nestjs/common';
import {
  DiagnosisLayer,
  RankedFinding,
  RuleFinding,
} from '../../domain/types/diagnosis.types';

@Injectable()
export class RankingService {
  private readonly layerPriority: Record<DiagnosisLayer, number> = {
    user_action: 9,
    ui_state: 2,
    frontend_app: 5,
    bff: 7,
    api: 8,
    domain: 6,
    db: 4,
    external: 3,
  };

  rank(findings: RuleFinding[]): RankedFinding[] {
    const clusterSeen = new Map<string, number>();

    const ranked = findings.map((finding) => {
      const baseScore = Math.round((finding.confidence ?? 0) * 100);
      const evidenceScore = Math.min((finding.evidenceIds?.length ?? 0) * 2, 10);
      const layerScore = this.layerPriority[finding.layer] ?? 0;

      const seenCount = clusterSeen.get(finding.cluster) ?? 0;
      clusterSeen.set(finding.cluster, seenCount + 1);

      const clusterPenalty = seenCount === 0 ? 0 : -10 * seenCount;
      const symptomPenalty = finding.isSymptomOnly ? -20 : 0;

      const finalScore =
        baseScore +
        evidenceScore +
        layerScore +
        clusterPenalty +
        symptomPenalty;

      const rankedFinding: RankedFinding = {
        ...finding,
        rankScore: {
          baseScore,
          evidenceScore,
          layerScore,
          clusterPenalty,
          symptomPenalty,
          finalScore,
        },
      };

      return rankedFinding;
    });

    ranked.sort((a, b) => {
      const rootVsSymptom =
        Number(Boolean(a.isSymptomOnly)) - Number(Boolean(b.isSymptomOnly));
      if (rootVsSymptom !== 0) {
        return rootVsSymptom;
      }

      const finalScoreDiff = b.rankScore.finalScore - a.rankScore.finalScore;
      if (finalScoreDiff !== 0) {
        return finalScoreDiff;
      }

      return b.confidence - a.confidence;
    });

    return ranked;
  }
}
```

# 九、DiagnosisConclusionService

> 规则：
- `topCause` 取第一个非 symptom finding
- `supportingCauses` 取其余非 symptom findings
- `symptoms` 收拢 symptom findings
- diagnosisState 由 `topCause.confidence` 决定，不看 finalScore

---

## `src/diagnosis/application/services/diagnosis-conclusion.service.ts`

```ts
import { Injectable } from '@nestjs/common';
import {
  DiagnosisConclusion,
  RankedFinding,
} from '../../domain/types/diagnosis.types';

@Injectable()
export class DiagnosisConclusionService {
  buildConclusion(rankedFindings: RankedFinding[]): DiagnosisConclusion {
    const rootCandidates = rankedFindings.filter((x) => !x.isSymptomOnly);
    const symptoms = rankedFindings.filter((x) => x.isSymptomOnly);

    const topCause = rootCandidates[0] ?? null;
    const supportingCauses = rootCandidates.slice(1);

    const diagnosisState = this.resolveDiagnosisState(topCause?.confidence);

    return {
      topCause,
      supportingCauses,
      symptoms,
      diagnosisState,
      summary: this.buildSummary(topCause, supportingCauses, symptoms),
    };
  }

  private resolveDiagnosisState(
    confidence?: number,
  ): 'confirmed' | 'probable' | 'inconclusive' {
    if (confidence === undefined || confidence === null) {
      return 'inconclusive';
    }

    if (confidence >= 0.85) {
      return 'confirmed';
    }

    if (confidence >= 0.6) {
      return 'probable';
    }

    return 'inconclusive';
  }

  private buildSummary(
    topCause: RankedFinding | null,
    supportingCauses: RankedFinding[],
    symptoms: RankedFinding[],
  ): string {
    if (!topCause) {
      if (symptoms.length > 0) {
        return `Only symptom-level findings were observed (${symptoms
          .map((x) => x.ruleCode)
          .join(', ')}), no root cause is confirmed.`;
      }

      return 'No actionable diagnosis finding was concluded.';
    }

    const supportingPart =
      supportingCauses.length > 0
        ? ` Supporting causes: ${supportingCauses
            .map((x) => x.ruleCode)
            .join(', ')}.`
        : '';

    const symptomPart =
      symptoms.length > 0
        ? ` Symptoms: ${symptoms.map((x) => x.ruleCode).join(', ')}.`
        : '';

    return `Top cause is ${topCause.ruleCode} (${topCause.title}).${supportingPart}${symptomPart}`;
  }
}
```

# 十、DominoChainBuilder

> 目标：
- root / supporting / symptom 节点分层
- 相同 `type + layer + cluster` 聚合
- edge 形成 root -> supporting -> symptom 的基础链路

---

## `src/diagnosis/application/builders/domino-chain.builder.ts`

```ts
import { Injectable } from '@nestjs/common';
import {
  DiagnosisConclusion,
  DominoChain,
  DominoEdge,
  DominoNode,
  RankedFinding,
} from '../../domain/types/diagnosis.types';

@Injectable()
export class DominoChainBuilder {
  build(conclusion: DiagnosisConclusion): DominoChain {
    const nodeMap = new Map<string, DominoNode>();

    const rootNode = conclusion.topCause
      ? this.upsertNode(nodeMap, 'root', conclusion.topCause)
      : null;

    const supportingNodes = conclusion.supportingCauses.map((finding) =>
      this.upsertNode(nodeMap, 'supporting', finding),
    );

    const symptomNodes = conclusion.symptoms.map((finding) =>
      this.upsertNode(nodeMap, 'symptom', finding),
    );

    const edges: DominoEdge[] = [];

    if (rootNode) {
      for (const node of supportingNodes) {
        if (node.id !== rootNode.id) {
          edges.push({
            from: rootNode.id,
            to: node.id,
            relation: 'contributes_to',
          });
        }
      }
    }

    if (supportingNodes.length > 0) {
      for (const symptomNode of symptomNodes) {
        edges.push({
          from: supportingNodes[supportingNodes.length - 1].id,
          to: symptomNode.id,
          relation: 'leads_to',
        });
      }
    } else if (rootNode) {
      for (const symptomNode of symptomNodes) {
        edges.push({
          from: rootNode.id,
          to: symptomNode.id,
          relation: 'explains',
        });
      }
    }

    return {
      nodes: [...nodeMap.values()],
      edges: this.dedupeEdges(edges),
    };
  }

  private upsertNode(
    nodeMap: Map<string, DominoNode>,
    type: 'root' | 'supporting' | 'symptom',
    finding: RankedFinding,
  ): DominoNode {
    const id = `node:${type}:${finding.layer}:${finding.cluster}`;
    const existing = nodeMap.get(id);

    if (existing) {
      existing.findings.push(finding);
      return existing;
    }

    const node: DominoNode = {
      id,
      type,
      title: this.resolveNodeTitle(type, finding),
      layer: finding.layer,
      findings: [finding],
    };

    nodeMap.set(id, node);
    return node;
  }

  private resolveNodeTitle(
    type: 'root' | 'supporting' | 'symptom',
    finding: RankedFinding,
  ): string {
    if (type === 'root') {
      return `Root: ${finding.title}`;
    }

    if (type === 'supporting') {
      return `Supporting: ${finding.title}`;
    }

    return `Symptom: ${finding.title}`;
  }

  private dedupeEdges(edges: DominoEdge[]): DominoEdge[] {
    const map = new Map<string, DominoEdge>();

    for (const edge of edges) {
      const key = `${edge.from}->${edge.to}:${edge.relation}`;
      map.set(key, edge);
    }

    return [...map.values()];
  }
}
```

## `src/diagnosis/application/builders/explanation.builder.ts`

```ts
import { Injectable } from '@nestjs/common';
import {
  DiagnosisConclusion,
  DiagnosisContext,
} from '../../domain/types/diagnosis.types';

@Injectable()
export class ExplanationBuilder {
  build(
    conclusion: DiagnosisConclusion,
    context: DiagnosisContext,
  ): string {
    const lines: string[] = [];

    lines.push(
      `Diagnosis trace ${context.traceId} on target ${context.targetId}.`,
    );

    if (conclusion.topCause) {
      lines.push(
        `Top cause: ${conclusion.topCause.title} (${conclusion.topCause.ruleCode}), confidence ${Math.round(
          conclusion.topCause.confidence * 100,
        )}%.`,
      );
    } else {
      lines.push('No root cause was confidently identified.');
    }

    if (conclusion.supportingCauses.length > 0) {
      lines.push(
        `Supporting causes: ${conclusion.supportingCauses
          .map((x) => `${x.title} (${x.ruleCode})`)
          .join('; ')}.`,
      );
    }

    if (conclusion.symptoms.length > 0) {
      lines.push(
        `Symptoms: ${conclusion.symptoms
          .map((x) => `${x.title} (${x.ruleCode})`)
          .join('; ')}.`,
      );
    }

    if (context.evidence.length > 0) {
      lines.push(
        `Evidence considered: ${context.evidence
          .slice(0, 5)
          .map((x) => x.label)
          .join('; ')}.`,
      );
    }

    lines.push(`Diagnosis state: ${conclusion.diagnosisState}.`);

    return lines.join(' ');
  }
}
```

## `src/diagnosis/application/services/diagnosis-query.service.ts`

```ts
import { Inject, Injectable, NotFoundException } from '@nestjs/common';
import { DIAGNOSIS_TASK_REPOSITORY } from '../../diagnosis.constants';
import { DiagnosisTaskRepository } from '../ports/diagnosis-task.repository';
import { DiagnosisContextBuilder } from '../builders/diagnosis-context.builder';
import { RuleEngineService } from '../../domain/rule-engine/rule-engine.service';
import { RankingService } from './ranking.service';
import { DiagnosisConclusionService } from './diagnosis-conclusion.service';
import { DominoChainBuilder } from '../builders/domino-chain.builder';
import { ExplanationBuilder } from '../builders/explanation.builder';

@Injectable()
export class DiagnosisQueryService {
  constructor(
    @Inject(DIAGNOSIS_TASK_REPOSITORY)
    private readonly repository: DiagnosisTaskRepository,
    private readonly contextBuilder: DiagnosisContextBuilder,
    private readonly ruleEngine: RuleEngineService,
    private readonly rankingService: RankingService,
    private readonly conclusionService: DiagnosisConclusionService,
    private readonly dominoChainBuilder: DominoChainBuilder,
    private readonly explanationBuilder: ExplanationBuilder,
  ) {}

  async getById(id: string) {
    const task = await this.repository.findById(id);

    if (!task) {
      throw new NotFoundException(`Diagnosis task not found: ${id}`);
    }

    if (task.status === 'completed' && task.result) {
      return task;
    }

    const context = this.contextBuilder.build(task);
    const findings = this.ruleEngine.evaluate(context);
    const rankedFindings = this.rankingService.rank(findings);
    const conclusion = this.conclusionService.buildConclusion(rankedFindings);
    const dominoChain = this.dominoChainBuilder.build(conclusion);
    const explanation = this.explanationBuilder.build(conclusion, context);

    return this.repository.saveResult(task.id, {
      conclusion,
      dominoChain,
      explanation,
    });
  }
}
```
