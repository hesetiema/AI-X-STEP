- 统一断言 symptom-only / dedupe

```ts
import { RuleFinding } from '../../src/diagnosis/domain/types/diagnosis.types';

export function getRuleCodes(findings: RuleFinding[]): string[] {
  return findings.map((x) => x.ruleCode);
}

export function expectRuleCodesToContain(
  findings: RuleFinding[],
  expectedCodes: string[],
): void {
  expect(getRuleCodes(findings)).toEqual(
    expect.arrayContaining(expectedCodes),
  );
}

export function expectRuleCodeHit(
  findings: RuleFinding[],
  expectedCode: string,
): void {
  expect(getRuleCodes(findings)).toContain(expectedCode);
}

export function expectRuleCodeMiss(
  findings: RuleFinding[],
  unexpectedCode: string,
): void {
  expect(getRuleCodes(findings)).not.toContain(unexpectedCode);
}

export function expectSymptomRule(
  findings: RuleFinding[],
  ruleCode: string,
): void {
  const finding = findings.find((x) => x.ruleCode === ruleCode);

  expect(finding).toBeDefined();
  expect(finding?.isSymptomOnly).toBe(true);
}

export function expectNonSymptomRule(
  findings: RuleFinding[],
  ruleCode: string,
): void {
  const finding = findings.find((x) => x.ruleCode === ruleCode);

  expect(finding).toBeDefined();
  expect(finding?.isSymptomOnly).toBe(false);
}

export function expectSingleFindingForRule(
  findings: RuleFinding[],
  ruleCode: string,
): void {
  const matched = findings.filter((x) => x.ruleCode === ruleCode);
  expect(matched).toHaveLength(1);
}

export function expectFindingMeta(
  findings: RuleFinding[],
  ruleCode: string,
  expected: Partial<Pick<RuleFinding, 'layer' | 'cluster' | 'isSymptomOnly'>>,
): void {
  const finding = findings.find((x) => x.ruleCode === ruleCode);

  expect(finding).toBeDefined();

  if (expected.layer !== undefined) {
    expect(finding?.layer).toBe(expected.layer);
  }

  if (expected.cluster !== undefined) {
    expect(finding?.cluster).toBe(expected.cluster);
  }

  if (expected.isSymptomOnly !== undefined) {
    expect(finding?.isSymptomOnly).toBe(expected.isSymptomOnly);
  }
}
```

- 改 `rule-engine-test.factory.ts`
- 少量补 spec case

不用每个 integration spec 都堆一长串 `new XxxRule()`。

---

## 2. assertions 收拢了重复断言
后续你最常写的就变成：

```ts
expectRuleCodeHit(findings, 'R201');
expectSymptomRule(findings, 'R501');
expectFindingMeta(findings, 'R301', {
  layer: 'db',
  cluster: 'db_timeout',
});
```

比手写：

```ts
expect(findings.map(x => x.ruleCode)).toContain(...)
expect(findings.find(...)?....).toBe(...)
```

要统一很多。

---

## 3. 保持 recall-first 边界清晰
这个 spec 只关心：

- 命中了谁
- 是否 symptom
- 基本 meta 是否对

不关心：

- 排序先后
- top cause
- finalScore
- diagnosisState

这些继续留在：
- `ranking.service.spec.ts`
- `diagnosis-conclusion.service.spec.ts`
