1. **规则文件更短**

## `src/diagnosis/domain/rule-engine/matchers/network.matcher.ts`

```ts
import { DiagnosisContext } from '../../types/diagnosis.types';

type EvidenceItem = DiagnosisContext['evidence'][number];

function toRecord(value: unknown): Record<string, unknown> | undefined {
  if (!value || typeof value !== 'object' || Array.isArray(value)) {
    return undefined;
  }

  return value as Record<string, unknown>;
}

function toLowerText(value: unknown): string {
  return String(value ?? '').toLowerCase();
}

function toNumber(value: unknown): number | null {
  return typeof value === 'number' && Number.isFinite(value) ? value : null;
}

export function getNetworkEvidence(context: DiagnosisContext): EvidenceItem[] {
  return context.evidence.filter((ev) => ev.type === 'network_error');
}

export function isHttp5xxEvidence(ev: EvidenceItem): boolean {
  const value = toRecord(ev.value);
  const status = toNumber(value?.status);

  return status !== null && status >= 500;
}

export function isTimeoutEvidence(
  ev: EvidenceItem,
  options?: {
    minDurationMs?: number;
  },
): boolean {
  const minDurationMs = options?.minDurationMs ?? 3000;
  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const code = toLowerText(value?.code);
  const durationMs = toNumber(value?.durationMs);

  return (
    label.includes('timeout') ||
    code === 'etimedout' ||
    (durationMs !== null && durationMs >= minDurationMs)
  );
}

export function isGatewayLikeEvidence(ev: EvidenceItem): boolean {
  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const status = toNumber(value?.status);

  return (
    label.includes('gateway') ||
    label.includes('bff') ||
    value?.gateway === true ||
    status === 502 ||
    status === 504
  );
}

export function collectNetworkEvidence(
  context: DiagnosisContext,
  predicate: (ev: EvidenceItem) => boolean,
): EvidenceItem[] {
  return getNetworkEvidence(context).filter(predicate);
}
```

```ts
import { DiagnosisContext } from '../../types/diagnosis.types';

type EvidenceItem = DiagnosisContext['evidence'][number];

function toRecord(value: unknown): Record<string, unknown> | undefined {
  if (!value || typeof value !== 'object' || Array.isArray(value)) {
    return undefined;
  }

  return value as Record<string, unknown>;
}

function toLowerText(value: unknown): string {
  return String(value ?? '').toLowerCase();
}

function toNumber(value: unknown): number | null {
  return typeof value === 'number' && Number.isFinite(value) ? value : null;
}

export function isLoadingEvidence(
  ev: EvidenceItem,
  options?: {
    minLoadingMs?: number;
  },
): boolean {
  const minLoadingMs = options?.minLoadingMs ?? 3000;
  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const loadingMs = toNumber(value?.loadingMs);

  return (
    ev.type === 'ui_state' &&
    (label.includes('loading') ||
      (loadingMs !== null && loadingMs >= minLoadingMs))
  );
}

export function hasLoadingSymptomFlag(context: DiagnosisContext): boolean {
  return context.symptoms.some((s) => toLowerText(s).includes('loading'));
}

export function isErrorToastEvidence(ev: EvidenceItem): boolean {
  if (ev.type !== 'ui_state') {
    return false;
  }

  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const toastType = toLowerText(value?.toastType);

  return label.includes('error toast') || toastType === 'error';
}

export function isEmptyStateEvidence(ev: EvidenceItem): boolean {
  if (ev.type !== 'ui_state') {
    return false;
  }

  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);

  return (
    label.includes('empty state') ||
    (value?.empty === true &&
      typeof value?.expectedCount === 'number' &&
      Number(value.expectedCount) > 0)
  );
}

export function isRetryLoopEvidence(ev: EvidenceItem): boolean {
  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const retryCount =
    typeof value?.retryCount === 'number' ? Number(value.retryCount) : 0;

  return label.includes('retry loop') || value?.loop === true || retryCount >= 3;
}

export function collectUiEvidence(
  context: DiagnosisContext,
  predicate: (ev: EvidenceItem) => boolean,
): EvidenceItem[] {
  return context.evidence.filter(predicate);
}
```

```ts
    return undefined;
  }

  return value as Record<string, unknown>;

function toLowerText(value: unknown): string {
  return String(value ?? '').toLowerCase();
}

function toNumber(value: unknown): number | null {

}

export function isExternalDependencyFailureEvidence(ev: EvidenceItem): boolean {
  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const status = toNumber(value?.status);

  return (
    value?.external === true &&
    (label.includes('provider') ||
      label.includes('third-party') ||
      label.includes('unavailable') ||
      status === 503 ||
      status === 504)
  );
}

export function isThirdPartyRateLimitEvidence(ev: EvidenceItem): boolean {
  const label = toLowerText(ev.label);
  const value = toRecord(ev.value);
  const status = toNumber(value?.status);

  return (
    value?.external === true &&
    (status === 429 ||
      label.includes('rate limit') ||
      label.includes('too many requests'))
  );
}

export function collectExternalEvidence(
  context: DiagnosisContext,
  predicate: (ev: EvidenceItem) => boolean,
): EvidenceItem[] {
  return context.evidence.filter(predicate);
}
```

## 1）`src/diagnosis/domain/rule-engine/rules/r201-api-5xx.rule.ts`

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisRule } from '../interfaces/diagnosis-rule.interface';
import { DiagnosisContext, RuleFinding } from '../../types/diagnosis.types';
import {
  collectNetworkEvidence,
  isHttp5xxEvidence,
} from '../matchers/network.matcher';

@Injectable()
export class R201Api5xxRule implements DiagnosisRule {
  readonly code = 'R201';

  evaluate(context: DiagnosisContext): RuleFinding[] {
    const matchedEvidence = collectNetworkEvidence(context, isHttp5xxEvidence);

    if (matchedEvidence.length === 0) {
      return [];
    }

    return [
      {
        ruleCode: 'R201',
        title: 'API 5xx root cause',
        summary:
          'Observed API 5xx response indicating backend/API-side instability.',
        confidence: 0.91,
        score: 91,
        layer: 'api',
        cluster: 'api_failure',
        evidenceIds: matchedEvidence.map((x) => x.id),
        isSymptomOnly: false,
        detail: {
          matchedCount: matchedEvidence.length,
        },
        rule: {
          code: 'R201',
          name: 'API 5xx indicates downstream instability',
          cluster: 'api_failure',
          layer: 'api',
          severity: 'high',
          isSymptomOnly: false,
        },
      },
    ];
  }
}
```

# 七、一个很实用的进一步抽象

如果你后面要继续减少重复，可以再加一个公共 helper，比如：

## `buildRuleFinding` factory
把所有 rule 返回结构里重复的部分收敛掉：

```ts
buildRuleFindingFromMatchedEvidence({
  ruleCode: 'R201',
  title: 'API 5xx root cause',
  summary: '...',
  confidence: 0.91,
  layer: 'api',
  cluster: 'api_failure',
  isSymptomOnly: false,
  matchedEvidence,
  severity: 'high',
  ruleName: 'API 5xx indicates downstream instability',
})
```

这样 rule 文件会再短一半。  
不过我建议你先完成 matcher 层，再做 finding factory，不然变化太多不利于你当前增量 hardening。
