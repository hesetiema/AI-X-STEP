## `src/modules/diagnosis/__tests__/e2e/in-memory-diagnosis.repository.ts`

```ts
export interface InMemoryDiagnosisTask {
  id: string;
  mode: 'inspect_diagnosis' | 'click_diagnosis';
  target: Record<string, any>;
  createdAt: string;
}

export interface InMemoryDiagnosisResult {
  diagnosisId: string;
  status: 'completed' | 'failed' | 'pending';
  createdAt: string;
  updatedAt: string;
  context: any;
  findings: any[];
  conclusion: any;
  explanation: any;
  dominoChain: any;
  errorMessage?: string;
}

export class InMemoryDiagnosisRepository {
  private readonly tasks = new Map<string, InMemoryDiagnosisTask>();
  private readonly results = new Map<string, InMemoryDiagnosisResult>();

  async createTask(task: InMemoryDiagnosisTask): Promise<void> {
    this.tasks.set(task.id, task);
  }

  async findTaskById(id: string): Promise<InMemoryDiagnosisTask | null> {
    return this.tasks.get(id) ?? null;
  }

  async findResultByDiagnosisId(
    diagnosisId: string,
  ): Promise<InMemoryDiagnosisResult | null> {
    return this.results.get(diagnosisId) ?? null;
  }

  async saveResult(result: InMemoryDiagnosisResult): Promise<void> {
    this.results.set(result.diagnosisId, result);
  }

  clear(): void {
    this.tasks.clear();
    this.results.clear();
  }

  seedTask(task: InMemoryDiagnosisTask): void {
    this.tasks.set(task.id, task);
  }

  seedResult(result: InMemoryDiagnosisResult): void {
    this.results.set(result.diagnosisId, result);
  }

  snapshot() {
    return {
      tasks: Array.from(this.tasks.values()),
      results: Array.from(this.results.values()),
    };
  }
}
```

# 5. 半真实 query service

同样，如果你已有实现，可对照校正。  
重点是符合你现在既定策略：

- 找不到 task → 404
- 有缓存结果 → 直接返回
- 无缓存 → lazy execute
- 抛错时直接向上抛，不保存 failed

## `src/modules/diagnosis/application/services/diagnosis-query.service.ts`

```ts
import { Inject, Injectable, NotFoundException } from '@nestjs/common';

import { DIAGNOSIS_REPOSITORY } from '../../infrastructure/constants/diagnosis-repository.token';
import { RuleEngineService } from '../../domain/services/rule-engine.service';
import { DiagnosisConclusionService } from '../../domain/services/diagnosis-conclusion.service';
import { ExplanationBuilder } from '../../domain/builders/explanation.builder';
import { DominoChainBuilder } from '../../domain/builders/domino-chain.builder';

@Injectable()
export class DiagnosisQueryService {
  constructor(
    @Inject(DIAGNOSIS_REPOSITORY)
    private readonly repository: {
      findTaskById(id: string): Promise<any | null>;
      findResultByDiagnosisId(id: string): Promise<any | null>;
      saveResult(result: any): Promise<void>;
    },
    @Inject('DiagnosisContextBuilder')
    private readonly contextBuilder: {
      build(task: any): Promise<any>;
    },
    private readonly ruleEngineService: RuleEngineService,
    private readonly diagnosisConclusionService: DiagnosisConclusionService,
    private readonly explanationBuilder: ExplanationBuilder,
    private readonly dominoChainBuilder: DominoChainBuilder,
  ) {}

  async getById(id: string) {
    const task = await this.repository.findTaskById(id);
    if (!task) {
      throw new NotFoundException('Diagnosis task not found');
    }

    const cachedResult = await this.repository.findResultByDiagnosisId(id);
    if (cachedResult) {
      return cachedResult;
    }

    const context = await this.contextBuilder.build(task);
    const findings = await this.ruleEngineService.run(context);
    const conclusion = await this.diagnosisConclusionService.conclude(
      findings,
      context,
    );
    const explanation = await this.explanationBuilder.build(
      conclusion,
      context,
    );
    const dominoChain = await this.dominoChainBuilder.build(
      conclusion,
      context,
    );

    const result = {
      diagnosisId: id,
      status: 'completed',
      createdAt: task.createdAt,
      updatedAt: new Date().toISOString(),
      context,
      findings,
      conclusion,
      explanation,
      dominoChain,
    };

    await this.repository.saveResult(result);

    return result;
  }
}
```
