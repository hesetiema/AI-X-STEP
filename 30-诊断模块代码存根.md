```ts
export function isNil(value: unknown): boolean {

export function isFalsyButValid(value: unknown): boolean {
  return value === 0 || value === false || value === '';
}

export function safeToString(value: unknown): string {
  if (value === null) return 'null';
  if (value === undefined) return 'undefined';
  return String(value);
}

export function isFallbackValue(
  value: unknown,
  fallbackTokens: string[] = [],
): boolean {
  if (value === null || value === undefined) {
    return false;
  }

  return fallbackTokens.includes(String(value));
}
```

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisRepository } from './diagnosis.repository';
import { DiagnosisTask } from '../../domain/models/diagnosis-task.model';
import { DiagnosisResult } from '../../domain/models/diagnosis-result.model';

@Injectable()
export class InMemoryDiagnosisRepository implements DiagnosisRepository {
  private readonly tasks = new Map<string, DiagnosisTask>();
  private readonly results = new Map<string, DiagnosisResult>();

  async createTask(task: DiagnosisTask): Promise<void> {
    this.tasks.set(task.id, task);
  }

  async findTaskById(id: string): Promise<DiagnosisTask | null> {
    return this.tasks.get(id) ?? null;
  }

  async findResultByDiagnosisId(id: string): Promise<DiagnosisResult | null> {
    return this.results.get(id) ?? null;
  }

  async saveResult(result: DiagnosisResult): Promise<void> {
    this.results.set(result.diagnosisId, result);
  }
}
```

## `src/modules/diagnosis/application/diagnosis-command.service.ts`

```ts
import { Inject, Injectable } from '@nestjs/common';
import { CreateDiagnosisDto } from './dto/create-diagnosis.dto';
import { DiagnosisTask } from '../domain/models/diagnosis-task.model';
import { DiagnosisStatus } from '../domain/enums/diagnosis-status.enum';
import { DIAGNOSIS_REPOSITORY } from '../constants/diagnosis-tokens';
import { DiagnosisRepository } from '../infrastructure/repositories/diagnosis.repository';

@Injectable()
export class DiagnosisCommandService {
  constructor(
    @Inject(DIAGNOSIS_REPOSITORY)
    private readonly diagnosisRepository: DiagnosisRepository,
  ) {}

  async createTask(dto: CreateDiagnosisDto): Promise<DiagnosisTask> {
    const task: DiagnosisTask = {
      id: this.generateDiagnosisId(),
      mode: dto.mode as any,
      status: DiagnosisStatus.PENDING,
      target: {
        pageUrl: dto.pageUrl,
        domSelector: dto.targetDomSelector,
        componentName: dto.targetComponent,
      },
      interactionId: dto.interactionId,
      lineageId: dto.lineageId,
      traceId: dto.traceId,
      snapshotId: dto.snapshotId,
      createdAt: new Date().toISOString(),
    };

    await this.diagnosisRepository.createTask(task);

    return task;
  }

  private generateDiagnosisId(): string {
    return `diag_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`;
  }
}
```

## `src/modules/diagnosis/application/diagnosis-query.service.ts`

```ts
import { Inject, Injectable, NotFoundException } from '@nestjs/common';
import { DIAGNOSIS_REPOSITORY } from '../constants/diagnosis-tokens';
import { DiagnosisRepository } from '../infrastructure/repositories/diagnosis.repository';
import { EvidenceCollectorService } from '../domain/services/evidence-collector.service';
import { ContextBuilderService } from '../domain/services/context-builder.service';
import { RuleEngineService } from '../domain/services/rule-engine.service';
import { DiagnosisConclusionService } from '../domain/services/diagnosis-conclusion.service';
import { ExplanationBuilder } from '../domain/builders/explanation-builder';
import { DominoChainBuilder } from '../domain/builders/domino-chain-builder';
import { DiagnosisResult } from '../domain/models/diagnosis-result.model';
import { DiagnosisStatus } from '../domain/enums/diagnosis-status.enum';

@Injectable()
export class DiagnosisQueryService {
  constructor(
    @Inject(DIAGNOSIS_REPOSITORY)
    private readonly diagnosisRepository: DiagnosisRepository,
    private readonly evidenceCollector: EvidenceCollectorService,
    private readonly contextBuilder: ContextBuilderService,
    private readonly ruleEngineService: RuleEngineService,
    private readonly diagnosisConclusionService: DiagnosisConclusionService,
    private readonly explanationBuilder: ExplanationBuilder,
    private readonly dominoChainBuilder: DominoChainBuilder,
  ) {}

  async getDiagnosisResult(diagnosisId: string): Promise<DiagnosisResult> {
    const task = await this.diagnosisRepository.findTaskById(diagnosisId);
    if (!task) {
      throw new NotFoundException(`Diagnosis task not found: ${diagnosisId}`);
    }

    const cached = await this.diagnosisRepository.findResultByDiagnosisId(
      diagnosisId,
    );
    if (cached) {
      return cached;
    }

    const evidence = await this.evidenceCollector.collect(task);
    const context = this.contextBuilder.build(task, evidence);
    const findings = this.ruleEngineService.run(context);
    const conclusion = this.diagnosisConclusionService.conclude(
      findings,
      context,
    );
    const explanation = this.explanationBuilder.build(
      context,
      conclusion,
      findings,
    );
    const dominoChain = this.dominoChainBuilder.build(
      context,
      conclusion,
      findings,
    );

    const result: DiagnosisResult = {
      diagnosisId,
      status: DiagnosisStatus.COMPLETED,
      context,
      findings,
      conclusion,
      explanation,
      dominoChain,
    };

    await this.diagnosisRepository.saveResult(result);

    return result;
  }
}
```

```ts
import { Body, Controller, Get, Param, Post } from '@nestjs/common';
import { DiagnosisCommandService } from '../application/diagnosis-command.service';
import { DiagnosisQueryService } from '../application/diagnosis-query.service';
import { CreateDiagnosisDto } from '../application/dto/create-diagnosis.dto';

@Controller('/api/v1/diagnosis')
export class DiagnosisController {
  constructor(
    private readonly diagnosisCommandService: DiagnosisCommandService,
    private readonly diagnosisQueryService: DiagnosisQueryService,
  ) {}

  @Post()
  async createDiagnosisTask(@Body() dto: CreateDiagnosisDto) {
    const task = await this.diagnosisCommandService.createTask(dto);

    return {
      id: task.id,
      status: task.status,
      mode: task.mode,
      createdAt: task.createdAt,
    };
  }

  @Get(':id')
  async getDiagnosisResult(@Param('id') id: string) {
    return this.diagnosisQueryService.getDiagnosisResult(id);
  }
}
```

```ts
import { Injectable } from '@nestjs/common';
import { RuleFinding } from '../models/rule-finding.model';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { RankedFinding } from '../models/ranked-finding.model';
import { countEvidenceRefs } from '../utils/diagnosis-evidence.utils';
import { getLayerPriority } from '../utils/diagnosis-layer.utils';
import { RANKING_WEIGHTS } from '../../constants/diagnosis.constants';

@Injectable()
export class RankingService {
  rank(findings: RuleFinding[], _context: DiagnosisContext): RankedFinding[] {
    const clusterLeaderMap = new Map<string, string>();

    const preScored = findings.map((finding) => {
      const baseConfidence = finding.confidence ?? 0;
      const rootCauseBonus = finding.isSymptomOnly
        ? 0
        : RANKING_WEIGHTS.rootCauseBonus;
      const symptomPenalty = finding.isSymptomOnly
        ? RANKING_WEIGHTS.symptomPenalty
        : 0;
      const upstreamBonus = getLayerPriority(finding.layer) / 1000;
      const evidenceScore =
        countEvidenceRefs(finding.evidenceRefs) * RANKING_WEIGHTS.evidencePerRef;
      const clusterBonus = finding.cluster ? RANKING_WEIGHTS.clusterBonus : 0;

      const provisional =
        baseConfidence +
        rootCauseBonus +
        upstreamBonus +
        evidenceScore +
        clusterBonus -
        symptomPenalty;

      return {
        finding,
        provisional,
      };
    });

    for (const item of [...preScored].sort((a, b) => b.provisional - a.provisional)) {
      if (item.finding.cluster && !clusterLeaderMap.has(item.finding.cluster)) {
        clusterLeaderMap.set(item.finding.cluster, item.finding.ruleCode);
      }
    }

    return preScored
      .map(({ finding, provisional }) => {
        const duplicatePenalty =
          finding.cluster &&
          clusterLeaderMap.get(finding.cluster) !== finding.ruleCode
            ? RANKING_WEIGHTS.duplicatePenalty
            : 0;

        const finalScore = provisional - duplicatePenalty;
        const rankReasons: string[] = [];

        if (!finding.isSymptomOnly) rankReasons.push('根因规则优先');
        if (finding.isSymptomOnly) rankReasons.push('症状规则已降权');
        if (getLayerPriority(finding.layer) >= 85) {
          rankReasons.push('上游 API 层优先级最高');
        } else if (getLayerPriority(finding.layer) >= 60) {
          rankReasons.push('中上游链路优先');
        }

        if (finding.cluster) {
          rankReasons.push('规则簇明确，可归入稳定问题类型');
        }

        if (countEvidenceRefs(finding.evidenceRefs) >= 3) {
          rankReasons.push('证据链完整度高');
        }

        if (finding.confidence >= 0.9) {
          rankReasons.push('规则置信度高');
        }

        if (duplicatePenalty > 0) {
          rankReasons.push('同 cluster 规则已做去重降权');
        }

        const ranked: RankedFinding = {
          ...finding,
          rankScore: {
            baseConfidence: finding.confidence ?? 0,
            rootCauseBonus: finding.isSymptomOnly
              ? 0
              : RANKING_WEIGHTS.rootCauseBonus,
            symptomPenalty: finding.isSymptomOnly
              ? RANKING_WEIGHTS.symptomPenalty
              : 0,
            upstreamBonus: getLayerPriority(finding.layer) / 1000,
            duplicatePenalty,
            evidenceScore:
              countEvidenceRefs(finding.evidenceRefs) *
              RANKING_WEIGHTS.evidencePerRef,
            clusterBonus: finding.cluster ? RANKING_WEIGHTS.clusterBonus : 0,
            finalScore,
          },
          rankReasons,
        };

        return ranked;
      })
      .sort((a, b) => b.rankScore.finalScore - a.rankScore.finalScore);
  }
}
```

```ts
import { Injectable } from '@nestjs/common';
import { RuleFinding } from '../models/rule-finding.model';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { DiagnosisConclusion } from '../models/diagnosis-conclusion.model';
import { DiagnosisState } from '../enums/diagnosis-state.enum';
import { RankingService } from './ranking.service';
import { MAX_REPAIR_HINTS, MAX_SUPPORTING_FINDINGS } from '../../constants/diagnosis.constants';

@Injectable()
export class DiagnosisConclusionService {
  constructor(private readonly rankingService: RankingService) {}

  conclude(
    findings: RuleFinding[],
    context: DiagnosisContext,
  ): DiagnosisConclusion {
    const ranked = this.rankingService.rank(findings, context);
    const topFinding = ranked[0] ?? null;

    const supportingFindings = ranked
      .filter(
        (item, index) =>
          index > 0 &&
          !item.isSymptomOnly &&
          item.cluster !== topFinding?.cluster,
      )
      .slice(0, MAX_SUPPORTING_FINDINGS);

    const symptomFindings = ranked.filter((item) => item.isSymptomOnly);

    const diagnosisState = this.resolveDiagnosisState(topFinding, context);

    const repairHints = [
      ...(topFinding?.suggestions ?? []),
      ...supportingFindings.flatMap((item) => item.suggestions ?? []),
    ]
      .filter(Boolean)
      .filter((value, index, array) => array.indexOf(value) === index)
      .slice(0, MAX_REPAIR_HINTS);

    return {
      topFinding,
      supportingFindings,
      symptomFindings,
      diagnosisState,
      summary: this.buildSummary(topFinding, diagnosisState),
      repairHints: repairHints.length
        ? repairHints
        : this.buildFallbackRepairHints(diagnosisState),
      scoreBreakdown: ranked.map((item) => ({
        ruleCode: item.ruleCode,
        finalScore: item.rankScore.finalScore,
        reason: item.rankReasons,
      })),
    };
  }

  private resolveDiagnosisState(topFinding: any, context: DiagnosisContext): DiagnosisState {
    if (topFinding) {
      return topFinding.confidence >= 0.9
        ? DiagnosisState.CONFIRMED_ROOT_CAUSE
        : DiagnosisState.PROBABLE_ROOT_CAUSE;
    }

    const weakEvidence =
      !context.responseSuccess &&
      !context.storeUpdated &&
      !context.renderTriggered;

    return weakEvidence
      ? DiagnosisState.INSUFFICIENT_EVIDENCE
      : DiagnosisState.NO_RULE_MATCHED;
  }

  private buildSummary(topFinding: any, state: DiagnosisState): string {
    if (topFinding) {
      return `已定位到根因：${topFinding.diagnosisLabel}。`;
    }

    if (state === DiagnosisState.INSUFFICIENT_EVIDENCE) {
      return '当前证据不足，无法确认明确根因。';
    }

    return '现有规则未识别出明确异常。';
  }

  private buildFallbackRepairHints(state: DiagnosisState): string[] {
    if (state === DiagnosisState.INSUFFICIENT_EVIDENCE) {
      return ['补充接口响应快照', '补充状态更新轨迹', '补充渲染证据'];
    }

    return ['检查业务计算逻辑', '补充更细的规则覆盖'];
  }
}
```

```ts
@Injectable()
```

```ts
  private buildSummaryText(

    conclusion: DiagnosisConclusion,
  ): string {
    const top = conclusion.topFinding;

    if (!top) {
      if (conclusion.diagnosisState === 'insufficient_evidence') {
        return '当前证据不足，暂时无法确认明确根因，建议补充接口、状态和渲染链路证据。';

      return '现有规则暂未识别出明确根因，建议结合更多业务规则与链路证据继续排查。';
    }

    if (context.mode === 'click_diagnosis') {
      return `已定位到可疑根因：${top.diagnosisLabel}。当前点击链路未按预期推进到后续阶段。`;
    }

    return `已定位到可疑根因：${top.diagnosisLabel}。当前链路中上游值为 ${String(
      context.apiValue ?? context.storeValue ?? context.selectorValue ?? '未知',

      context.domValue ?? context.displayedValue ?? '未知',
    )}。`;
  }

  private buildEvidenceNarrative(
    context: DiagnosisContext,
    conclusion: DiagnosisConclusion,
    _findings: RuleFinding[],
  ): string[] {
    const lines: string[] = [];

    if (context.mode === 'click_diagnosis') {
      if (context.clickDetected) lines.push('检测到 click 事件已发生。');
      if (context.handlerStarted === false) lines.push('未观察到 handler 正常启动。');
      if (context.requestSent === false) lines.push('后续 request 未发出。');

      for (const item of conclusion.symptomFindings ?? []) {
        lines.push(`外显现象：${item.diagnosisLabel}。`);
      }

      return lines.length ? lines : ['当前点击链路证据有限。'];
    }

    if (context.apiFieldPath) {

        `接口字段 ${context.apiFieldPath} 当前值为 ${String(context.apiValue)}。`,

  private buildOperatorAdvice(
    conclusion: DiagnosisConclusion,
    _findings: RuleFinding[],

      ...(conclusion.repairHints ?? []),
      ...(conclusion.topFinding?.suggestions ?? []),
      ...(conclusion.supportingFindings ?? []).flatMap((item) => item.suggestions ?? []),
      ...(conclusion.symptomFindings ?? []).flatMap((item) => item.suggestions ?? []),
    ]);

    return merged.slice(0, MAX_REPAIR_HINTS);
  }
}
```

## `src/modules/diagnosis/domain/builders/domino-chain-builder.ts`

```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisContext } from '../models/diagnosis-context.model';
import { DiagnosisConclusion } from '../models/diagnosis-conclusion.model';
import { RuleFinding } from '../models/rule-finding.model';
import { DominoChainNode } from '../models/domino-chain-node.model';
import { DominoNodeStatus } from '../enums/domino-node-status.enum';
import { dedupeEvidenceRefs } from '../utils/diagnosis-evidence.utils';
import { normalizeFindingLayerToDominoLayer } from '../utils/diagnosis-layer.utils';

@Injectable()
export class DominoChainBuilder {
  build(
    context: DiagnosisContext,
    conclusion: DiagnosisConclusion,
    findings: RuleFinding[],
  ): DominoChainNode[] {
    const layers =
      context.mode === 'click_diagnosis'
        ? ['click', 'handler', 'request', 'response', 'state', 'render', 'dom']
        : ['response', 'state', 'selector', 'render', 'dom'];

    const nodeMap = new Map<string, DominoChainNode>();

    for (const layer of layers) {
      nodeMap.set(layer, {
        id: `node_${layer}`,
        layer,
        label: this.buildLabel(layer),
        value: this.extractValue(layer, context),
        status: DominoNodeStatus.OK,
        relatedRuleCodes: [],
        evidenceRefs: [],
      });
    }

    const topRuleCode = conclusion.topFinding?.ruleCode;
    const supportingCodes = new Set(
      (conclusion.supportingFindings ?? []).map((item) => item.ruleCode),
    );
    const symptomCodes = new Set(
      (conclusion.symptomFindings ?? []).map((item) => item.ruleCode),
    );

    for (const finding of findings) {
      const layer = normalizeFindingLayerToDominoLayer(finding.layer);
      const node = nodeMap.get(layer);
      if (!node) continue;

      node.relatedRuleCodes = [...(node.relatedRuleCodes ?? []), finding.ruleCode];
      node.evidenceRefs = [...(node.evidenceRefs ?? []), ...(finding.evidenceRefs ?? [])];

      if (finding.ruleCode === topRuleCode) {
        node.isRootCause = true;
        node.status = DominoNodeStatus.ERROR;
      } else if (supportingCodes.has(finding.ruleCode)) {
        node.isSupporting = true;
        if (node.status === DominoNodeStatus.OK) {
          node.status = DominoNodeStatus.WARNING;
        }
      } else if (symptomCodes.has(finding.ruleCode)) {
        node.isSymptom = true;
        if (node.status === DominoNodeStatus.OK) {
          node.status = DominoNodeStatus.SYMPTOM;
        }
      }
    }

    return [...nodeMap.values()].map((node) => ({
      ...node,
      relatedRuleCodes: [...new Set(node.relatedRuleCodes ?? [])],
      evidenceRefs: dedupeEvidenceRefs(node.evidenceRefs),
    }));
  }

  private buildLabel(layer: string): string {
    switch (layer) {
      case 'click':
        return 'Click';
      case 'handler':
        return 'Handler';
      case 'request':
        return 'Request';
      case 'response':
        return 'Response';
      case 'state':
        return 'State';
      case 'selector':
        return 'Selector';
      case 'render':
        return 'Render';
      case 'dom':
        return 'DOM';
      default:
        return layer;
    }
  }

  private extractValue(layer: string, context: DiagnosisContext): unknown {
    switch (layer) {
      case 'click':
        return context.clickDetected;
      case 'handler':
        return context.handlerStarted;
      case 'request':
        return context.requestSent;
      case 'response':
        return context.apiValue;
      case 'state':
        return context.storeValue;
      case 'selector':
        return context.selectorValue;
      case 'render':
        return context.renderOutputValue;
      case 'dom':
        return context.domValue ?? context.displayedValue;
      default:
        return undefined;
    }
  }
}
```

# 13. 现在这套 stub 已经能做什么

这套代码已经能让你先跑通这条链：

```text
POST /api/v1/diagnosis
GET /api/v1/diagnosis/:id
```

行为是：

1. 创建 diagnosis task
2. 查询结果时如果没缓存则 lazy execute
3. 用默认 evidence/context 生成一个可诊断上下文
4. rule engine 跑出 findings
5. ranking + conclusion
6. explanation + domino chain
7. repository 缓存结果并返回

也就是说，这已经不是“纯空壳”，而是一个 **可演化 MVP starter**。
