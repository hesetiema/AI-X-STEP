# 1) `test/unit/rules/r201-api-5xx.rule.spec.ts`

```ts
import { R201Api5xxRule } from '../../../src/diagnosis/domain/rule-engine/rules/r201-api-5xx.rule';
import { buildDiagnosisContext } from '../../helpers/diagnosis-context.fixture';
import {
  expectRuleMiss,
  expectSingleRuleHit,
} from '../../helpers/rule-unit-assertions';

describe('R201Api5xxRule', () => {
  let rule: R201Api5xxRule;

  beforeEach(() => {
    rule = new R201Api5xxRule();
  });

  it('should hit when network evidence contains 5xx status', () => {
    const context = buildDiagnosisContext({
      evidence: [
        {
          id: 'ev-api-1',
          type: 'network_error',
          label: 'POST /orders 500',
          value: { status: 500, url: '/api/orders' },
          source: 'network-observer',
        },
      ],
    });

    const result = rule.evaluate(context);

    expectSingleRuleHit(result, {
      ruleCode: 'R201',
      layer: 'api',
      cluster: 'api_failure',
      isSymptomOnly: false,
      evidenceIds: ['ev-api-1'],
    });
  });

  it('should not hit when status is non-5xx', () => {
    const context = buildDiagnosisContext({
      evidence: [
        {
          id: 'ev-api-2',
          type: 'network_error',
          label: 'POST /orders 400',
          value: { status: 400, url: '/api/orders' },
          source: 'network-observer',
        },
      ],
    });

    const result = rule.evaluate(context);

    expectRuleMiss(result);
  });
});
```

# 6) `test/unit/rules/r501-ui-loading-symptom.rule.spec.ts`

`R501` 是多入口命中规则，所以保留 3 个 case 更合适。

```ts
import { R501UiLoadingSymptomRule } from '../../../src/diagnosis/domain/rule-engine/rules/r501-ui-loading-symptom.rule';
import { buildDiagnosisContext } from '../../helpers/diagnosis-context.fixture';
import {
  expectRuleMiss,
  expectSingleRuleHit,
} from '../../helpers/rule-unit-assertions';

describe('R501UiLoadingSymptomRule', () => {
  let rule: R501UiLoadingSymptomRule;

  beforeEach(() => {
    rule = new R501UiLoadingSymptomRule();
  });

  it('should hit when loading evidence exists', () => {
    const context = buildDiagnosisContext({
      symptoms: [],
      evidence: [
        {
          id: 'ev-ui-1',
          type: 'ui_state',
          label: 'loading > 12s',
          value: { loadingMs: 12000 },
          source: 'frontend-sdk',
        },
      ],
    });

    const result = rule.evaluate(context);

    expectSingleRuleHit(result, {
      ruleCode: 'R501',
      layer: 'ui_state',
      cluster: 'ui_loading',
      isSymptomOnly: true,
      evidenceIds: ['ev-ui-1'],
      detail: {
        matchedCount: 1,
        symptomMatched: false,
      },
    });
  });

  it('should hit when loading symptom flag exists even without loading evidence', () => {
    const context = buildDiagnosisContext({
      symptoms: ['ui_loading'],
      evidence: [
        {
          id: 'ev-ui-2',
          type: 'ui_event',
          label: 'click submit',
          value: { element: '#submit-btn' },
          source: 'frontend-sdk',
        },
      ],
    });

    const result = rule.evaluate(context);

    expectSingleRuleHit(result, {
      ruleCode: 'R501',
      layer: 'ui_state',
      cluster: 'ui_loading',
      isSymptomOnly: true,
      evidenceIds: [],
      detail: {
        matchedCount: 0,
        symptomMatched: true,
      },
    });
  });

  it('should not hit when both loading evidence and loading symptom flag are absent', () => {
    const context = buildDiagnosisContext({
      symptoms: ['submit_failed'],
      evidence: [
        {
          id: 'ev-ui-3',
          type: 'ui_event',
          label: 'click submit',
          value: { element: '#submit-btn' },
          source: 'frontend-sdk',
        },
      ],
    });

    const result = rule.evaluate(context);

    expectRuleMiss(result);
  });
});
```
