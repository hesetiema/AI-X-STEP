```ts
import { Injectable } from '@nestjs/common';
import { DiagnosisRule } from '../interfaces/diagnosis-rule.interface';
import {
  DiagnosisContext,
  RuleFinding,
} from '../../types/diagnosis.types';

@Injectable()
export class R102UserActionInvalidTargetRule implements DiagnosisRule {
  readonly code = 'R102';

  evaluate(context: DiagnosisContext): RuleFinding[] {
    const matchedEvidence = context.evidence.filter((ev) => {
      if (ev.type !== 'ui_event') {
        return false;
      }

      const value = ev.value as Record<string, unknown> | undefined;
      const expectedTargetId = String(
        value?.expectedTargetId ?? context.targetId,
      );
      const actualTargetId = String(
        value?.actualTargetId ?? value?.element ?? '',
      );

      if (!actualTargetId) {
        return false;
      }

      return actualTargetId !== expectedTargetId;
    });

    if (matchedEvidence.length === 0) {
      return [];
    }

    return [
      {
        ruleCode: 'R102',
        title: 'Invalid user action target',
        summary:
          'User action was captured on a target inconsistent with the expected diagnosis target.',
        confidence: 0.81,
        score: 81,
        layer: 'user_action',
        cluster: 'invalid_action_target',
        evidenceIds: matchedEvidence.map((x) => x.id),
        isSymptomOnly: false,
        detail: {
          expectedTargetId: context.targetId,
        },
        rule: {
          code: 'R102',
          name: 'User action target mismatch',
          cluster: 'invalid_action_target',
          layer: 'user_action',
          severity: 'high',
          isSymptomOnly: false,
        },
      },
    ];

}
```

```ts
      return [];
    }

    return [
      {
        ruleCode: 'R504',
        title: 'UI retry loop symptom',
        summary:
          'The UI is repeatedly retrying, indicating unstable upstream dependency or unresolved failure.',
        confidence: 0.71,
        score: 71,
        layer: 'ui_state',
        cluster: 'ui_retry_loop',
        evidenceIds: matchedEvidence.map((x) => x.id),
        isSymptomOnly: true,
        detail: {
          matchedCount: matchedEvidence.length,
        },
        rule: {
          code: 'R504',
          name: 'Repeated UI retry loop',
          cluster: 'ui_retry_loop',
          layer: 'ui_state',
          severity: 'medium',
          isSymptomOnly: true,
        },
      },
    ];
  }
}
```
