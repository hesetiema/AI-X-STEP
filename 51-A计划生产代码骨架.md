```ts
import { DiagnosisTaskEntity } from './diagnosis-test-types';

export interface DiagnosisTaskRepository {
  create(input: Record<string, unknown>): Promise<DiagnosisTaskEntity>;
  findById(id: string): Promise<DiagnosisTaskEntity | null>;
  saveResult(
    id: string,
    result: DiagnosisTaskEntity['result'],
  ): Promise<DiagnosisTaskEntity>;
}

export class FakeDiagnosisTaskRepository implements DiagnosisTaskRepository {
  private readonly store = new Map<string, DiagnosisTaskEntity>();

  async create(input: Record<string, unknown>): Promise<DiagnosisTaskEntity> {
    const now = new Date().toISOString();
    const entity: DiagnosisTaskEntity = {
      id: `diag-${this.store.size + 1}`,
      status: 'pending',
      input,
      createdAt: now,
      updatedAt: now,
    };

    this.store.set(entity.id, entity);
    return entity;
  }

  async findById(id: string): Promise<DiagnosisTaskEntity | null> {
    return this.store.get(id) ?? null;
  }

  async saveResult(
    id: string,
    result: DiagnosisTaskEntity['result'],
  ): Promise<DiagnosisTaskEntity> {
    const entity = this.store.get(id);
    if (!entity) {
      throw new Error(`Diagnosis task not found: ${id}`);
    }

    const updated: DiagnosisTaskEntity = {
      ...entity,
      status: 'completed',
      result,
      updatedAt: new Date().toISOString(),
    };

    this.store.set(id, updated);
    return updated;
  }

  seed(entity: DiagnosisTaskEntity): void {
    this.store.set(entity.id, entity);
  }

  clear(): void {
    this.store.clear();
  }
}
```

```ts
    );
  });

    const input = [
      buildRuleFinding({
        ruleCode: 'R301',

        cluster: 'db_timeout',
        confidence: 0.8,
      }),
      buildRuleFinding({
        ruleCode: 'R201',
        layer: 'api',
        cluster: 'api_failure',
        confidence: 0.8,
      }),

    const result = service.rank(input);

    expect(result[0].ruleCode).toBe('R201');
  });

  it('should apply cluster dedupe penalty to lower duplicated cluster findings', () => {
    const input = [
      buildRuleFinding({
        ruleCode: 'R201',
        layer: 'api',
        cluster: 'api_failure',
        confidence: 0.9,
      }),
      buildRuleFinding({
        ruleCode: 'R202',
        layer: 'api',
        cluster: 'api_failure',
        confidence: 0.88,
      }),
      buildRuleFinding({
        ruleCode: 'R301',
        layer: 'db',
        cluster: 'db_timeout',
        confidence: 0.84,
      }),
    ];

    const result = service.rank(input);

    expect(result[0].ruleCode).toBe('R201');
    expect(result.map((x) => x.ruleCode)).toContain('R301');

    const duplicateClusterFinding = result.find((x) => x.ruleCode === 'R202');
    const primaryClusterFinding = result.find((x) => x.ruleCode === 'R201');

    expect(primaryClusterFinding).toBeDefined();
    expect(duplicateClusterFinding).toBeDefined();
    expect(primaryClusterFinding!.rankScore.clusterPenalty).toBeGreaterThanOrEqual(0);
    expect(duplicateClusterFinding!.rankScore.clusterPenalty).toBeLessThanOrEqual(0);
```

## `DiagnosisQueryService`

```ts
import { Inject, NotFoundException } from '@nestjs/common';
import { DIAGNOSIS_TASK_REPOSITORY } from '../../diagnosis.constants';

export class DiagnosisQueryService {
  constructor(
    @Inject(DIAGNOSIS_TASK_REPOSITORY)
    private readonly repository: any,
    private readonly contextBuilder: any,
    private readonly ruleEngine: any,
    private readonly rankingService: any,
    private readonly conclusionService: any,
    private readonly dominoChainBuilder: any,
    private readonly explanationBuilder: any,
  ) {}

  async getById(id: string) {
    const task = await this.repository.findById(id);

    if (!task) {
      throw new NotFoundException(`Diagnosis task not found: ${id}`);
    }

    if (task.status === 'completed' && task.result) {
      return task;
    }

    const context = this.contextBuilder.build(task);
    const findings = this.ruleEngine.evaluate(context);
    const rankedFindings = this.rankingService.rank(findings);
    const conclusion = this.conclusionService.buildConclusion(rankedFindings);
    const dominoChain = this.dominoChainBuilder.build(conclusion);
    const explanation = this.explanationBuilder.build(conclusion, context);

    return this.repository.saveResult(task.id, {
      conclusion,
      dominoChain,
      explanation,
    });
  }
}
```
