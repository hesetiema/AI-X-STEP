```ts
import { DEFAULT_FALLBACK_TOKENS } from './constants';
import { DiagnosisContext } from './types';

export function getFallbackTokens(ctx: DiagnosisContext): string[] {
  return ctx.fallbackTokens?.length ? ctx.fallbackTokens : DEFAULT_FALLBACK_TOKENS;
}

export function isFallbackValue(value: unknown, ctx: DiagnosisContext): boolean {
  if (typeof value !== 'string') return false;
  return getFallbackTokens(ctx).includes(value);
}

export function isNil(value: unknown): boolean {
  return value === null || value === undefined;
}

export function isEmptyCollection(value: unknown): boolean {
  if (Array.isArray(value)) return value.length === 0;
  if (typeof value === 'object' && value !== null) return Object.keys(value).length === 0;
  return false;
}

export function hasMeaningfulValue(value: unknown): boolean {
  if (isNil(value)) return false;
  if (value === '') return false;
  if (isEmptyCollection(value)) return false;
  return true;
}

export function isValidFalsyValue(value: unknown): boolean {
  return value === 0 || value === false || value === '';
}

export function bool(v: unknown): boolean {
  return Boolean(v);
}
```

## 5.1 ranking.ts

```ts
import { ROOT_CAUSE_PRIORITY, SPECIFICITY_ORDER } from './constants';
import { DiagnosisContext, RuleFinding } from './types';

export function getRuleCluster(ruleCode: string): string {
  const map: Record<string, string> = {
    R001: 'api_missing_or_empty',
    R002: 'api_missing_or_empty',
    R003: 'response_consumption',
    R101: 'state_update',
    R102: 'render_subscription',
    R103: 'selector_mapping',
    R201: 'formatter_fallback',
    R202: 'formatter_fallback',
    R203: 'render_trigger',
    R204: 'render_output',
    R301: 'dom_update',
    R302: 'dom_visibility',
    R303: 'dom_fallback',
    R401: 'click_handler',
    R402: 'handler_effect',
    R403: 'request_ui_chain',
  };

  return map[ruleCode] ?? ruleCode;
}

export function deduplicateFindings(findings: RuleFinding[]): RuleFinding[] {
  const sorted = [...findings].sort((a, b) => {
    const sa = SPECIFICITY_ORDER[a.ruleCode] ?? 0;
    const sb = SPECIFICITY_ORDER[b.ruleCode] ?? 0;
    if (sb !== sa) return sb - sa;
    return b.confidence - a.confidence;
  });

  const seenClusters = new Set<string>();
  const kept: RuleFinding[] = [];

  for (const finding of sorted) {
    const cluster = finding.cluster ?? getRuleCluster(finding.ruleCode);
    if (seenClusters.has(cluster)) continue;
    seenClusters.add(cluster);
    kept.push({ ...finding, cluster });
  }

  return kept;
}

export function rerankFindings(findings: RuleFinding[], _ctx: DiagnosisContext): RuleFinding[] {
  return [...findings].sort((a, b) => {
    const scoreA =
      (ROOT_CAUSE_PRIORITY[a.layer] ?? 0) * 0.4 + a.confidence * 100 * 0.6;
    const scoreB =
      (ROOT_CAUSE_PRIORITY[b.layer] ?? 0) * 0.4 + b.confidence * 100 * 0.6;
    return scoreB - scoreA;
  });
}
```

```ts
function normalizeContext(input: DiagnosisContext): DiagnosisContext {
  return {
    ...input,
    apiFieldExists: input.apiFieldExists ?? false,
    responseSuccess: input.responseSuccess ?? false,
    storeUpdated: input.storeUpdated ?? false,
    selectorRan: input.selectorRan ?? false,
    renderTriggered: input.renderTriggered ?? false,
    formatterOutputIsFallback: input.formatterOutputIsFallback ?? false,
    domUpdated: input.domUpdated ?? false,
    domVisible: input.domVisible ?? true,
    clickDetected: input.clickDetected ?? false,
    handlerStarted: input.handlerStarted ?? false,
    requestSent: input.requestSent ?? false,
    responseReceived: input.responseReceived ?? false,
    stateChanged: input.stateChanged ?? false,
    renderCommitted: input.renderCommitted ?? false,
    fallbackTokens: input.fallbackTokens?.length ? input.fallbackTokens : DEFAULT_FALLBACK_TOKENS,
    evidenceRefs: input.evidenceRefs ?? [],
  };
}

function runRuleGroup(ctx: DiagnosisContext, rules: RuleEvaluator[]): RuleFinding[] {
  const findings: RuleFinding[] = [];

  for (const rule of rules) {
    const finding = rule(ctx);
    if (finding) findings.push(finding);
  }

  return findings;
}

function buildSummary(findings: RuleFinding[], ctx: DiagnosisContext): string {
  if (!findings.length) {
    return ctx.mode === 'inspect_diagnosis'
      ? '当前证据不足，尚无法明确页面值异常发生在哪一层。'
      : '当前证据不足，尚无法明确点击后未生效的中断点。';
  }

  const top1 = findings[0];
  return `最可能的原因是：${top1.title}。${top1.summary}`;
}

export function runRuleEngine(input: DiagnosisContext): DiagnosisResult {
  const ctx = normalizeContext(input);

  let findings: RuleFinding[] = [];
  findings = findings.concat(runRuleGroup(ctx, dataSourceRules));
  findings = findings.concat(runRuleGroup(ctx, stateBindingRules));
  findings = findings.concat(runRuleGroup(ctx, renderTransformRules));
  findings = findings.concat(runRuleGroup(ctx, domRules));
  findings = findings.concat(runRuleGroup(ctx, interactionRules));

  findings = deduplicateFindings(findings);
  findings = rerankFindings(findings, ctx);

  return {
    top1: findings[0] ?? null,
    findings,
    summary: buildSummary(findings, ctx),
  };
}
```

```ts
export function walkDecisionTree(
  node: DecisionNode,
  ctx: DiagnosisContext,
  path: string[] = []
): { path: string[]; hitRuleCode?: string } {
  const currentPath = [...path, node.question];
  const matched = node.when(ctx);

  if (!matched) {
    if (!node.no) return { path: currentPath };
    return walkDecisionTree(node.no, ctx, currentPath);
  }

  if (node.hitRuleCode) {
    return {
      path: currentPath,
      hitRuleCode: node.hitRuleCode,
    };
  }

  if (!node.yes) {
    return { path: currentPath };
  }

  return walkDecisionTree(node.yes, ctx, currentPath);
}
```
